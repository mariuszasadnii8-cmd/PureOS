17macro_rules! err {
18    ($span:expr, $message:expr $(,)?) => {
19        Error::new($span.span(), $message).to_compile_error()
20    };
21    ($span:expr, $message:expr, $($args:expr),*) => {
22        Error::new($span.span(), format!($message, $($args),*)).to_compile_error()
23    };
24}

26/// Attribute macro for marking structs as UEFI protocols.
27///
28/// The macro can only be applied to a struct, and takes one argument, either a
29/// GUID string or the path to a `Guid` constant.
30///
31/// The macro implements the [`Protocol`] trait and the `unsafe` [`Identify`]
32/// trait for the struct. See the [`Protocol`] trait for details of how it is
33/// used.
34///
35/// # Safety
36///
37/// The caller must ensure that the correct GUID is attached to the
38/// type. An incorrect GUID could lead to invalid casts and other
39/// unsound behavior.
40///
41/// # Example
42///
43/// ```
44/// use uefi::{Guid, Identify, guid};
45/// use uefi::proto::unsafe_protocol;
46///
47/// #[unsafe_protocol("12345678-9abc-def0-1234-56789abcdef0")]
48/// struct ExampleProtocol1 {}
49///
50/// const PROTO_GUID: Guid = guid!("12345678-9abc-def0-1234-56789abcdef0");
51/// #[unsafe_protocol(PROTO_GUID)]
52/// struct ExampleProtocol2 {}
53///
54/// assert_eq!(ExampleProtocol1::GUID, PROTO_GUID);
55/// assert_eq!(ExampleProtocol2::GUID, PROTO_GUID);
56/// ```
57///
58/// [`Identify`]: https://docs.rs/uefi/latest/uefi/data_types/trait.Identify.html
59/// [`Protocol`]: https://docs.rs/uefi/latest/uefi/proto/trait.Protocol.html
60/// [send-and-sync]: https://doc.rust-lang.org/nomicon/send-and-sync.html
61#[proc_macro_attribute]
62pub fn unsafe_protocol(args: TokenStream, input: TokenStream) -> TokenStream {
63    let expr = parse_macro_input!(args as Expr);
64
65    let guid_val = match expr {
66        Expr::Lit(ExprLit {
67            lit: Lit::Str(lit), ..
68        }) => {
69            quote!(::uefi::guid!(#lit))
70        }
71        Expr::Path(ExprPath { path, .. }) => quote!(#path),
72        _ => {
73            return err!(
74                expr,
75                "macro input must be either a string literal or path to a constant"
76            )
77            .into()
78        }
79    };
80
81    let item_struct = parse_macro_input!(input as ItemStruct);
82
83    let ident = &item_struct.ident;
84    let (impl_generics, ty_generics, where_clause) = item_struct.generics.split_for_impl();
85
86    quote! {
87        #item_struct
88
89        unsafe impl #impl_generics ::uefi::Identify for #ident #ty_generics #where_clause {
90            const GUID: ::uefi::Guid = #guid_val;
91        }
92
93        impl #impl_generics ::uefi::proto::Protocol for #ident #ty_generics #where_clause {}
94    }
95    .into()
96}

98/// Custom attribute for a UEFI executable entry point.
99///
100/// This attribute modifies a function to mark it as the entry point for
101/// a UEFI executable. The function:
102/// * Must return [`Status`].
103/// * Must have zero parameters.
104/// * Can optionally be `unsafe`.
105///
106/// The global system table pointer and global image handle will be set
107/// automatically.
108///
109/// # Examples
110///
111/// ```no_run
112/// #![no_main]
113///
114/// use uefi::prelude::*;
115///
116/// #[entry]
117/// fn main() -> Status {
118///     Status::SUCCESS
119/// }
120/// ```
121///
122/// [`Status`]: https://docs.rs/uefi/latest/uefi/struct.Status.html
123#[proc_macro_attribute]
124pub fn entry(args: TokenStream, input: TokenStream) -> TokenStream {
125    // This code is inspired by the approach in this embedded Rust crate:
126    // https://github.com/rust-embedded/cortex-m-rt/blob/965bf1e3291571e7e3b34834864117dc020fb391/macros/src/lib.rs#L85
127
128    let mut errors = TokenStream2::new();
129
130    if !args.is_empty() {
131        errors.append_all(err!(
132            TokenStream2::from(args),
133            "Entry attribute accepts no arguments"
134        ));
135    }
136
137    let mut f = parse_macro_input!(input as ItemFn);
138
139    if let Some(ref abi) = f.sig.abi {
140        errors.append_all(err!(abi, "Entry function must have no ABI modifier"));
141    }
142    if let Some(asyncness) = f.sig.asyncness {
143        errors.append_all(err!(asyncness, "Entry function should not be async"));
144    }
145    if let Some(constness) = f.sig.constness {
146        errors.append_all(err!(constness, "Entry function should not be const"));
147    }
148    if !f.sig.generics.params.is_empty() {
149        errors.append_all(err!(
150            f.sig.generics.params,
151            "Entry function should not be generic"
152        ));
153    }
154    if !f.sig.inputs.is_empty() {
155        errors.append_all(err!(f.sig.inputs, "Entry function must have no arguments"));
156    }
157
158    // Show most errors all at once instead of one by one.
159    if !errors.is_empty() {
160        return errors.into();
161    }
162
163    let signature_span = f.sig.span();
164
165    // Fill in the image handle and system table arguments automatically.
166    let image_handle_ident = quote!(internal_image_handle);
167    let system_table_ident = quote!(internal_system_table);
168    f.sig.inputs = parse_quote_spanned!(
169        signature_span=>
170            #image_handle_ident: ::uefi::Handle,
171            #system_table_ident: *const ::core::ffi::c_void,
172    );
173
174    // Insert code at the beginning of the entry function to set the global
175    // image handle and system table pointer.
176    f.block.stmts.insert(
177        0,
178        parse_quote! {
179            unsafe {
180                ::uefi::boot::set_image_handle(#image_handle_ident);
181                ::uefi::table::set_system_table(#system_table_ident.cast());
182            }
183        },
184    );
185
186    // Set the required ABI.
187    f.sig.abi = Some(parse_quote_spanned!(signature_span=> extern "efiapi"));
188
189    // Strip any visibility modifiers.
190    f.vis = Visibility::Inherited;
191
192    let unsafety = &f.sig.unsafety;
193    let fn_ident = &f.sig.ident;
194    let fn_output = &f.sig.output;
195
196    // Get the expected argument types for the main function.
197    let expected_args = quote!(::uefi::Handle, *const core::ffi::c_void);
198
199    let fn_type_check = quote_spanned! {signature_span=>
200        // Cast from the function type to a function pointer with the same
201        // signature first, then try to assign that to an unnamed constant with
202        // the desired function pointer type.
203        //
204        // The cast is used to avoid an "expected fn pointer, found fn item"
205        // error if the signature is wrong, since that's not what we are
206        // interested in here. Instead we want to tell the user what
207        // specifically in the function signature is incorrect.
208        const _:
209            // The expected fn pointer type.
210            #unsafety extern "efiapi" fn(#expected_args) -> ::uefi::Status =
211            // Cast from a fn item to a function pointer.
212            #fn_ident as #unsafety extern "efiapi" fn(#expected_args) #fn_output;
213    };
214
215    let result = quote! {
216        #fn_type_check
217
218        #[export_name = "efi_main"]
219        #f
220
221    };
222    result.into()
223}
