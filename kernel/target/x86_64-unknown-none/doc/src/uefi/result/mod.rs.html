3//! Facilities for dealing with UEFI operation results.
4
5use core::fmt::Debug;
6
7/// The error type that we use, essentially a status code + optional additional data
8mod error;
9pub use error::Error;
10
11/// Definition of UEFI's standard status codes
12mod status;
13pub use status::{Status, StatusExt};
14
15/// Return type of most UEFI functions. Both success and error payloads are optional.
16///
17/// Almost all UEFI operations provide a status code as an output which
18/// indicates either success, a warning, or an error. This type alias maps
19/// [`Status::SUCCESS`] to the `Ok` variant (with optional `Output` data), and
20/// maps both warning and error statuses to the `Err` variant of type [`Error`],
21/// which may carry optional inner `ErrData`.
22///
23/// Warnings are treated as errors by default because they generally indicate
24/// an abnormal situation.
25///
26/// Some convenience methods are provided by the [`ResultExt`] trait.
27pub type Result<Output = (), ErrData = ()> = core::result::Result<Output, Error<ErrData>>;
28
29/// Extension trait which provides some convenience methods for [`Result`].
30pub trait ResultExt<Output, ErrData: Debug> {
31    /// Extract the UEFI status from this result
32    fn status(&self) -> Status;
33
34    /// Transform the ErrData value to ()
35    fn discard_errdata(self) -> Result<Output>;
36
37    /// Calls `op` if the result contains a warning, otherwise returns
38    /// the result unchanged.
39    ///
40    /// By default warning statuses are treated as errors (i.e. stored in the
41    /// `Err` variant) because they generally indicate an abnormal
42    /// situation. In rare cases though it may be helpful to handle a
43    /// warning. This method is similar to [`Result::or_else`], except that
44    /// `op` is called only when the status is a warning.
45    ///
46    /// # Example
47    ///
48    /// ```
49    /// use uefi::{Result, ResultExt, Status};
50    ///
51    /// # use uefi::StatusExt;
52    /// # fn x() -> uefi::Result {
53    /// # let some_result = Status::WARN_RESET_REQUIRED.to_result();
54    /// // Treat a specific warning as success, propagate others as errors.
55    /// some_result.handle_warning(|err| {
56    ///     if err.status() == Status::WARN_RESET_REQUIRED {
57    ///         Ok(())
58    ///     } else {
59    ///         Err(err)
60    ///     }
61    /// })?;
62    /// # Status::SUCCESS.to_result()
63    /// # }
64    /// ```
65    fn handle_warning<O>(self, op: O) -> Result<Output, ErrData>
66    where
67        O: FnOnce(Error<ErrData>) -> Result<Output, ErrData>;
68}
69
70impl<Output, ErrData: Debug> ResultExt<Output, ErrData> for Result<Output, ErrData> {
71    fn status(&self) -> Status {
72        match self {
73            Ok(_) => Status::SUCCESS,
74            Err(e) => e.status(),
75        }
76    }
77
78    fn discard_errdata(self) -> Result<Output> {
79        match self {
80            Ok(o) => Ok(o),
81            Err(e) => Err(e.status().into()),
82        }
83    }
84
85    fn handle_warning<O>(self, op: O) -> Self
86    where
87        O: FnOnce(Error<ErrData>) -> Self,
88    {
89        match self {
90            Ok(output) => Ok(output),
91            Err(err) => {
92                if err.status().is_warning() {
93                    op(err)
94                } else {
95                    Err(err)
96                }
97            }
98        }
99    }
100}
