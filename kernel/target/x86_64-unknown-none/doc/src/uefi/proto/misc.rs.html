12/// Protocol for retrieving a high-resolution timestamp counter.
13/// **Note:**
14/// If your UEFI firmware not support timestamp protocol which first added at UEFI spec 2.4 2013.
15/// you also could use `RDTSC` in rust, here is a demo [Slint-UI](https://github.com/slint-ui/slint/blob/2c0ba2bc0f151eba8d1fa17839fa2ac58832ca80/examples/uefi-demo/main.rs#L28-L62) who use uefi-rs.
16#[derive(Debug)]
17#[repr(transparent)]
18#[unsafe_protocol(TimestampProtocol::GUID)]
19pub struct Timestamp(TimestampProtocol);
20
21impl Timestamp {
22    /// Get the current value of the timestamp counter.
23    #[must_use]
24    pub fn get_timestamp(&self) -> u64 {
25        unsafe { (self.0.get_timestamp)() }
26    }
27
28    /// Get the properties of the timestamp counter.
29    pub fn get_properties(&self) -> Result<TimestampProperties> {
30        let mut properties = TimestampProperties::default();
31        unsafe { (self.0.get_properties)(&mut properties) }.to_result_with_val(|| properties)
32    }
33}

35/// Protocol to register for a notification when ResetSystem is called.
36#[derive(Debug)]
37#[repr(transparent)]
38#[unsafe_protocol(ResetNotificationProtocol::GUID)]
39pub struct ResetNotification(ResetNotificationProtocol);
40
41impl ResetNotification {
42    /// Register a notification function to be called when ResetSystem() is called.
43    ///
44    ///
45    /// # Example
46    ///
47    /// ```rust
48    /// use log::info;
49    /// use uefi::{boot, Handle};
50    /// use uefi::proto::misc::{ResetNotification};
51    /// use uefi_raw::Status;
52    /// use uefi_raw::table::runtime;
53    ///
54    ///
55    /// // value efi_reset_fn is the type of ResetSystemFn, a function pointer
56    /// unsafe extern "efiapi" fn efi_reset_fn(
57    ///         rt: runtime::ResetType,
58    ///         status: Status,
59    ///         data_size: usize,
60    ///         data: *const u8,
61    /// ){
62    ///     info!("Inside the event callback");
63    ///     info!("do what you want");
64    /// }
65    ///
66    /// pub fn test(image: Handle) {
67    ///
68    ///     let mut rn = boot::open_protocol_exclusive::<ResetNotification>(image)
69    ///         .expect("Failed to open Timestamp protocol");
70    ///
71    ///     rn.register_reset_notify(efi_reset_fn)
72    ///         .expect("Failed to register a reset notification function!");
73    /// }
74    /// ```
75    pub fn register_reset_notify(&mut self, reset_function: ResetSystemFn) -> Result {
76        unsafe { (self.0.register_reset_notify)(&mut self.0, reset_function) }.to_result()
77    }
78
79    /// Remove a reset notification function that was previously registered with [`ResetNotification::register_reset_notify`].
80    pub fn unregister_reset_notify(&mut self, reset_function: ResetSystemFn) -> Result {
81        unsafe { (self.0.unregister_reset_notify)(&mut self.0, reset_function) }.to_result()
82    }
83}
