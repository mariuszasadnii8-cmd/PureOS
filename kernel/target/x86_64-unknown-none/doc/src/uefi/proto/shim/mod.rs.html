36const SHA1_DIGEST_SIZE: usize = 20;
37const SHA256_DIGEST_SIZE: usize = 32;

39/// Authenticode hashes of some UEFI application.
40#[derive(Debug)]
41pub struct Hashes {
42    /// SHA256 Authenticode Digest
43    pub sha256: [u8; SHA256_DIGEST_SIZE],
44    /// SHA1 Authenticode Digest
45    pub sha1: [u8; SHA1_DIGEST_SIZE],
46}

48// These macros set the correct calling convention for the Shim protocol methods.
49
50#[cfg(target_arch = "x86")]
51macro_rules! shim_function {
52    (fn $args:tt -> $return_type:ty) => (extern "cdecl" fn $args -> $return_type)
53}
54
55#[cfg(target_arch = "x86_64")]
56macro_rules! shim_function {
57    (fn $args:tt -> $return_type:ty) => (extern "sysv64" fn $args -> $return_type)
58}
59
60#[cfg(any(target_arch = "arm", target_arch = "aarch64"))]
61macro_rules! shim_function {
62    (fn $args:tt -> $return_type:ty) => (extern "C" fn $args -> $return_type)
63}

65/// The Shim lock protocol.
66///
67/// This protocol is not part of the UEFI specification, but is
68/// installed by the [Shim bootloader](https://github.com/rhboot/shim)
69/// which is commonly used by Linux distributions to support UEFI
70/// Secure Boot. Shim is built with an embedded certificate that is
71/// used to validate another EFI application before running it. That
72/// application may itself be a bootloader that needs to validate
73/// another EFI application before running it, and the shim lock
74/// protocol exists to support that.
75#[derive(Debug)]
76#[repr(C)]
77#[unsafe_protocol("605dab50-e046-4300-abb6-3dd810dd8b23")]
78pub struct ShimLock {
79    verify: shim_function! { fn(buffer: *const u8, size: u32) -> Status },
80    hash: shim_function! {
81        fn(
82            buffer: *const u8,
83            size: u32,
84            context: *mut Context,
85            sha256: *mut [u8; SHA256_DIGEST_SIZE],
86            sha1: *mut [u8; SHA1_DIGEST_SIZE]
87        ) -> Status
88    },
89    context: shim_function! { fn(buffer: *const u8, size: u32, context: *mut Context) -> Status },
90}
91
92impl ShimLock {
93    /// Verify that an EFI application is signed by the certificate
94    /// embedded in shim.
95    ///
96    /// The buffer's size must fit in a `u32`; if that condition is not
97    /// met then a `BAD_BUFFER_SIZE` error will be returned and the shim
98    /// lock protocol will not be called.
99    pub fn verify(&self, buffer: &[u8]) -> Result {
100        let size: u32 = buffer
101            .len()
102            .try_into()
103            .map_err(|_| Error::from(Status::BAD_BUFFER_SIZE))?;
104        (self.verify)(buffer.as_ptr(), size).to_result()
105    }
106    /// Compute the Authenticode Hash of the provided EFI application.
107    ///
108    /// The buffer's size must fit in a `u32`; if that condition is not
109    /// met then a `BAD_BUFFER_SIZE` error will be returned and the shim
110    /// lock protocol will not be called.
111    pub fn hash(&self, buffer: &[u8], hashes: &mut Hashes) -> Result {
112        let ptr: *const u8 = buffer.as_ptr();
113        let size: u32 = buffer
114            .len()
115            .try_into()
116            .map_err(|_| Error::from(Status::BAD_BUFFER_SIZE))?;
117
118        let mut context = MaybeUninit::<Context>::uninit();
119        (self.context)(ptr, size, context.as_mut_ptr()).to_result()?;
120        (self.hash)(
121            ptr,
122            size,
123            context.as_mut_ptr(),
124            &mut hashes.sha256,
125            &mut hashes.sha1,
126        )
127        .to_result()
128    }
129}
