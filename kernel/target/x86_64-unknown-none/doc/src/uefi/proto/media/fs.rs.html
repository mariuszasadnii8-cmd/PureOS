3//! File system support protocols.
4
5use super::file::{Directory, FileHandle};
6use crate::proto::unsafe_protocol;
7use crate::{Result, StatusExt};
8use core::ptr;
9use uefi_raw::protocol::file_system::SimpleFileSystemProtocol;
10
11/// Allows access to a FAT-12/16/32 file system.
12///
13/// This interface is implemented by some storage devices
14/// to allow file access to the contained file systems.
15///
16/// # Accessing `SimpleFileSystem` protocol
17///
18/// Use [`boot::get_image_file_system`] to retrieve the `SimpleFileSystem`
19/// protocol associated with a given image handle.
20///
21/// See the [`boot`] documentation for more details of how to open a protocol.
22///
23/// [`boot::get_image_file_system`]: crate::boot::get_image_file_system
24/// [`boot`]: crate::boot#accessing-protocols
25#[derive(Debug)]
26#[repr(transparent)]
27#[unsafe_protocol(SimpleFileSystemProtocol::GUID)]
28pub struct SimpleFileSystem(SimpleFileSystemProtocol);
29
30impl SimpleFileSystem {
31    /// Open the root directory on a volume.
32    ///
33    /// # Errors
34    ///
35    /// See section `EFI_SIMPLE_FILE_SYSTEM_PROTOCOL.OpenVolume()` in the UEFI Specification
36    /// for more details.
37    ///
38    /// If you can't find the function definition, try searching for
39    /// `EFI_SIMPLE_FILE SYSTEM_PROTOCOL.OpenVolume()` (this has a space in between FILE and
40    /// SYSTEM; it could be a typo in the UEFI spec).
41    ///
42    /// * [`uefi::Status::UNSUPPORTED`]
43    /// * [`uefi::Status::NO_MEDIA`]
44    /// * [`uefi::Status::DEVICE_ERROR`]
45    /// * [`uefi::Status::VOLUME_CORRUPTED`]
46    /// * [`uefi::Status::ACCESS_DENIED`]
47    /// * [`uefi::Status::OUT_OF_RESOURCES`]
48    /// * [`uefi::Status::MEDIA_CHANGED`]
49    pub fn open_volume(&mut self) -> Result<Directory> {
50        let mut ptr = ptr::null_mut();
51        unsafe { (self.0.open_volume)(&mut self.0, &mut ptr) }
52            .to_result_with_val(|| unsafe { Directory::new(FileHandle::new(ptr.cast())) })
53    }
54}
