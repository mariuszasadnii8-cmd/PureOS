16/// Load File Protocol.
17///
18/// Used to obtain files, that are primarily boot options, from arbitrary
19/// devices.
20///
21/// # UEFI Spec Description
22/// The EFI_LOAD_FILE_PROTOCOL is a simple protocol used to obtain files from
23/// arbitrary devices.
24///
25/// When the firmware is attempting to load a file, it first attempts to use the
26/// device’s Simple File System protocol to read the file. If the file system
27/// protocol is found, the firmware implements the policy of interpreting the
28/// File Path value of the file being loaded. If the device does not support the
29/// file system protocol, the firmware then attempts to read the file via the
30/// EFI_LOAD_FILE_PROTOCOL and the LoadFile() function. In this case the
31/// LoadFile() function implements the policy of interpreting the File Path
32/// value.
33#[derive(Debug)]
34#[repr(transparent)]
35#[unsafe_protocol(LoadFileProtocol::GUID)]
36pub struct LoadFile(LoadFileProtocol);
37
38impl LoadFile {
39    /// Causes the driver to load a specified file.
40    ///
41    /// # Parameters
42    /// - `file_path` The device specific path of the file to load.
43    /// - `boot_policy` The [`BootPolicy`] to use.
44    ///
45    /// # Errors
46    /// - `uefi::status::EFI_SUCCESS` The file was loaded.
47    /// - `uefi::status::EFI_UNSUPPORTED` The device does not support the
48    ///   provided BootPolicy.
49    /// - `uefi::status::EFI_INVALID_PARAMETER` FilePath is not a valid device
50    ///   path, or BufferSize is NULL.
51    /// - `uefi::status::EFI_NO_MEDIA` No medium was present to load the file.
52    /// - `uefi::status::EFI_DEVICE_ERROR` The file was not loaded due to a
53    ///   device error.
54    /// - `uefi::status::EFI_NO_RESPONSE` The remote system did not respond.
55    /// - `uefi::status::EFI_NOT_FOUND` The file was not found.
56    /// - `uefi::status::EFI_ABORTED` The file load process was manually
57    ///   cancelled.
58    /// - `uefi::status::EFI_BUFFER_TOO_SMALL` The BufferSize is too small to
59    ///   read the current directory entry. BufferSize has been updated with the
60    ///   size needed to complete the request.
61    /// - `uefi::status::EFI_WARN_FILE_SYSTEM` The resulting Buffer contains
62    ///   UEFI-compliant file system.
63    ///
64    /// [`BootPolicy`]: uefi::proto::BootPolicy
65    #[cfg(feature = "alloc")]
66    #[allow(clippy::extra_unused_lifetimes)] // false positive, it is used
67    pub fn load_file<'a>(
68        &mut self,
69        file_path: &DevicePath,
70        boot_policy: BootPolicy,
71    ) -> Result<Box<[u8]>> {
72        let this = core::ptr::addr_of_mut!(*self).cast();
73
74        let fetch_data_fn = |buf: &'a mut [u8]| {
75            let mut size = buf.len();
76            let status = unsafe {
77                (self.0.load_file)(
78                    this,
79                    file_path.as_ffi_ptr().cast(),
80                    boot_policy.into(),
81                    &mut size,
82                    buf.as_mut_ptr().cast(),
83                )
84            };
85            status.to_result_with_err(|_| Some(size)).map(|_| buf)
86        };
87
88        #[cfg(not(feature = "unstable"))]
89        let file: Box<[u8]> = make_boxed::<[u8], _>(fetch_data_fn)?;
90
91        #[cfg(feature = "unstable")]
92        let file = make_boxed::<[u8], _, _>(fetch_data_fn, Global)?;
93
94        Ok(file)
95    }
96}

98/// Load File2 Protocol.
99///
100/// The Load File2 protocol is used to obtain files from arbitrary devices that
101/// are not boot options.
102///
103/// # UEFI Spec Description
104///
105/// The EFI_LOAD_FILE2_PROTOCOL is a simple protocol used to obtain files from
106/// arbitrary devices that are not boot options. It is used by LoadImage() when
107/// its BootOption parameter is FALSE and the FilePath does not have an instance
108/// of the EFI_SIMPLE_FILE_SYSTEM_PROTOCOL.
109#[derive(Debug)]
110#[repr(transparent)]
111#[unsafe_protocol(LoadFile2Protocol::GUID)]
112pub struct LoadFile2(LoadFile2Protocol);
113
114impl LoadFile2 {
115    /// Causes the driver to load a specified file.
116    ///
117    /// # Parameters
118    /// - `file_path` The device specific path of the file to load.
119    ///
120    /// # Errors
121    /// - `uefi::status::EFI_SUCCESS` The file was loaded.
122    /// - `uefi::status::EFI_UNSUPPORTED` BootPolicy is TRUE.
123    /// - `uefi::status::EFI_INVALID_PARAMETER` FilePath is not a valid device
124    ///   path, or BufferSize is NULL.
125    /// - `uefi::status::EFI_NO_MEDIA` No medium was present to load the file.
126    /// - `uefi::status::EFI_DEVICE_ERROR` The file was not loaded due to a
127    ///   device error.
128    /// - `uefi::status::EFI_NO_RESPONSE` The remote system did not respond.
129    /// - `uefi::status::EFI_NOT_FOUND` The file was not found.
130    /// - `uefi::status::EFI_ABORTED` The file load process was manually
131    ///   cancelled.
132    /// - `uefi::status::EFI_BUFFER_TOO_SMALL` The BufferSize is too small to
133    ///   read the current directory entry. BufferSize has been updated with the
134    ///   size needed to complete the request.
135    #[cfg(feature = "alloc")]
136    #[allow(clippy::extra_unused_lifetimes)] // false positive, it is used
137    pub fn load_file<'a>(&mut self, file_path: &DevicePath) -> Result<Box<[u8]>> {
138        let this = core::ptr::addr_of_mut!(*self).cast();
139
140        let fetch_data_fn = |buf: &'a mut [u8]| {
141            let mut size = buf.len();
142            let status = unsafe {
143                (self.0.load_file)(
144                    this,
145                    file_path.as_ffi_ptr().cast(),
146                    false, /* always false - see spec */
147                    &mut size,
148                    buf.as_mut_ptr().cast(),
149                )
150            };
151            status.to_result_with_err(|_| Some(size)).map(|_| buf)
152        };
153
154        #[cfg(not(feature = "unstable"))]
155        let file: Box<[u8]> = make_boxed::<[u8], _>(fetch_data_fn)?;
156
157        #[cfg(feature = "unstable")]
158        let file = make_boxed::<[u8], _, _>(fetch_data_fn, Global)?;
159
160        Ok(file)
161    }
162}
