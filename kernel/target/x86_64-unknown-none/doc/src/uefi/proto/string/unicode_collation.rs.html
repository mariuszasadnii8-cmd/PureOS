3//! The Unicode Collation Protocol.
4//!
5//! This protocol is used in the boot services environment to perform
6//! lexical comparison functions on Unicode strings for given languages.
7
8use crate::data_types::{CStr16, CStr8};
9use crate::proto::unsafe_protocol;
10use core::cmp::Ordering;
11use core::fmt::{self, Display, Formatter};
12use uefi_raw::protocol::string::UnicodeCollationProtocol;

14/// The Unicode Collation Protocol.
15///
16/// Used to perform case-insensitive comparisons of strings.
17#[derive(Debug)]
18#[repr(transparent)]
19#[unsafe_protocol(UnicodeCollationProtocol::GUID)]
20pub struct UnicodeCollation(UnicodeCollationProtocol);
21
22impl UnicodeCollation {
23    /// Performs a case insensitive comparison of two
24    /// null-terminated strings.
25    #[must_use]
26    pub fn stri_coll(&self, s1: &CStr16, s2: &CStr16) -> Ordering {
27        let order = unsafe { (self.0.stri_coll)(&self.0, s1.as_ptr().cast(), s2.as_ptr().cast()) };
28        order.cmp(&0)
29    }
30
31    /// Performs a case insensitive comparison between a null terminated
32    /// pattern string and a null terminated string.
33    ///
34    /// This function checks if character pattern described in `pattern`
35    /// is found in `string`. If the pattern match succeeds, true is returned.
36    /// Otherwise, false is returned.
37    ///
38    /// The following syntax can be used to build the string `pattern`:
39    ///
40    /// |Pattern Character            |Meaning                                           |
41    /// |-----------------------------|--------------------------------------------------|
42    /// |*                            | Match 0 or more characters                       |
43    /// |?                            | Match any one character                          |
44    /// |[`char1` `char2`...`charN`]| Match any character in the set                   |
45    /// |[`char1`-`char2`]          | Match any character between `char1` and `char2`|
46    /// |`char`                      | Match the character `char`                      |
47    ///
48    /// For example, the pattern "*.Fw" will match all strings that end
49    /// in ".FW", ".fw", ".Fw" or ".fW". The pattern "[a-z]" will match any
50    /// letter in the alphabet. The pattern "z" will match the letter "z".
51    /// The pattern "d?.*" will match the character "D" or "d" followed by
52    /// any single character followed by a "." followed by any string.
53    #[must_use]
54    pub fn metai_match(&self, s: &CStr16, pattern: &CStr16) -> bool {
55        unsafe { (self.0.metai_match)(&self.0, s.as_ptr().cast(), pattern.as_ptr().cast()) }
56    }
57
58    /// Converts the characters in `s` to lower case characters.
59    pub fn str_lwr<'a>(
60        &self,
61        s: &CStr16,
62        buf: &'a mut [u16],
63    ) -> Result<&'a CStr16, StrConversionError> {
64        let mut last_index = 0;
65        for (i, c) in s.iter().enumerate() {
66            *buf.get_mut(i).ok_or(StrConversionError::BufferTooSmall)? = (*c).into();
67            last_index = i;
68        }
69        *buf.get_mut(last_index + 1)
70            .ok_or(StrConversionError::BufferTooSmall)? = 0;
71
72        unsafe { (self.0.str_lwr)(&self.0, buf.as_mut_ptr()) };
73
74        Ok(unsafe { CStr16::from_u16_with_nul_unchecked(buf) })
75    }
76
77    /// Converts the characters in `s` to upper case characters.
78    pub fn str_upr<'a>(
79        &self,
80        s: &CStr16,
81        buf: &'a mut [u16],
82    ) -> Result<&'a CStr16, StrConversionError> {
83        let mut last_index = 0;
84        for (i, c) in s.iter().enumerate() {
85            *buf.get_mut(i).ok_or(StrConversionError::BufferTooSmall)? = (*c).into();
86            last_index = i;
87        }
88        *buf.get_mut(last_index + 1)
89            .ok_or(StrConversionError::BufferTooSmall)? = 0;
90
91        unsafe { (self.0.str_upr)(&self.0, buf.as_mut_ptr()) };
92
93        Ok(unsafe { CStr16::from_u16_with_nul_unchecked(buf) })
94    }

96    /// Converts the 8.3 FAT file name `fat` to a null terminated string.
97    pub fn fat_to_str<'a>(
98        &self,
99        fat: &CStr8,
100        buf: &'a mut [u16],
101    ) -> Result<&'a CStr16, StrConversionError> {
102        if buf.len() < fat.as_bytes().len() {
103            return Err(StrConversionError::BufferTooSmall);
104        }
105        unsafe {
106            (self.0.fat_to_str)(
107                &self.0,
108                fat.as_bytes().len(),
109                fat.as_ptr().cast(),
110                buf.as_mut_ptr(),
111            )
112        };
113        Ok(unsafe { CStr16::from_u16_with_nul_unchecked(buf) })
114    }
115
116    /// Converts the null terminated string `s` to legal characters in a FAT file name.
117    pub fn str_to_fat<'a>(
118        &self,
119        s: &CStr16,
120        buf: &'a mut [u8],
121    ) -> Result<&'a CStr8, StrConversionError> {
122        if s.as_slice_with_nul().len() > buf.len() {
123            return Err(StrConversionError::BufferTooSmall);
124        }
125        let failed = unsafe {
126            (self.0.str_to_fat)(
127                &self.0,
128                s.as_ptr().cast(),
129                s.as_slice_with_nul().len(),
130                buf.as_mut_ptr(),
131            )
132        };
133        if failed {
134            Err(StrConversionError::ConversionFailed)
135        } else {
136            // After the conversion, there is a possibility that the converted string
137            // is smaller than the original `s` string.
138            // When the converted string is smaller, there will be a bunch of trailing
139            // nulls.
140            // To remove all those trailing nulls:
141            let mut last_null_index = buf.len() - 1;
142            for i in (0..buf.len()).rev() {
143                if buf[i] != 0 {
144                    last_null_index = i + 1;
145                    break;
146                }
147            }
148            let buf = unsafe { core::slice::from_raw_parts(buf.as_ptr(), last_null_index + 1) };
149            Ok(unsafe { CStr8::from_bytes_with_nul_unchecked(buf) })
150        }
151    }
152}

154/// Errors returned by [`UnicodeCollation::str_lwr`] and [`UnicodeCollation::str_upr`].
155#[derive(Clone, Copy, Debug, PartialEq, Eq)]
156pub enum StrConversionError {
157    /// The conversion failed.
158    ConversionFailed,
159    /// The buffer given is too small to hold the string.
160    BufferTooSmall,
161}
162
163impl Display for StrConversionError {
164    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
165        write!(
166            f,
167            "{}",
168            match self {
169                Self::ConversionFailed => "conversion failed",
170                Self::BufferTooSmall => "buffer too small",
171            }
172        )
173    }
174}

176impl core::error::Error for StrConversionError {}
