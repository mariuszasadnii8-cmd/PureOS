3//! Protocols for converting between UEFI strings and [`DevicePath`]/[`DevicePathNode`].
4
5// Note on return types: the specification of the conversion functions
6// is a little unusual in that they return a pointer rather than
7// `EFI_STATUS`. A NULL pointer is used to indicate an error, and the
8// spec says that will only happen if the input pointer is null (which
9// can't happen here since we use references as input, not pointers), or
10// if there is insufficient memory. So we treat any NULL output as an
11// `OUT_OF_RESOURCES` error.
12
13use crate::mem::PoolAllocation;
14use crate::proto::device_path::{DevicePath, DevicePathNode};
15use crate::proto::unsafe_protocol;
16use crate::{CStr16, Char16, Result, Status};
17use core::ops::Deref;
18use core::ptr::NonNull;
19use uefi_raw::protocol::device_path::{DevicePathFromTextProtocol, DevicePathToTextProtocol};

21/// Parameter for [`DevicePathToText`] that alters the output format.
22///
23/// * `DisplayOnly(false)` produces parseable output.
24/// * `DisplayOnly(true)` produces output that _may_ be shorter and not
25///   parseable.
26///
27/// Example of how a node's text representation may be altered by this
28/// parameter:
29/// * `DisplayOnly(false)`: `Ata(Primary,Master,0x1)`
30/// * `DisplayOnly(true)`: `Ata(0x1)`
31#[derive(Clone, Copy, Debug)]
32pub struct DisplayOnly(pub bool);
33
34/// Parameter for [`DevicePathToText`] that alters the output format.
35///
36/// * `AllowShortcuts(false)`: node names are based only on the node's type and
37///   subtype.
38/// * `AllowShortcuts(true)` _may_ alter the node name based on other fields
39///   within the node.
40///
41/// Example of how a node's text representation may be altered by this
42/// parameter:
43/// * `AllowShortcuts(false)`: `VenMsg(E0C14753-F9BE-11D2-9A0C-0090273FC14D)`
44/// * `AllowShortcuts(true)`: `VenPcAnsi()`
45#[derive(Clone, Copy, Debug)]
46pub struct AllowShortcuts(pub bool);
47
48/// UCS-2 string allocated from UEFI pool memory.
49///
50/// This is similar to a [`CString16`], but used for memory that was allocated
51/// internally by UEFI rather than the Rust allocator.
52///
53/// [`CString16`]: crate::CString16
54#[derive(Debug)]
55pub struct PoolString(PoolAllocation);
56
57impl PoolString {
58    fn new(text: *const Char16) -> Result<Self> {
59        NonNull::new(text.cast_mut())
60            .map(|p| Self(PoolAllocation::new(p.cast())))
61            .ok_or(Status::OUT_OF_RESOURCES.into())
62    }
63}
64
65impl Deref for PoolString {
66    type Target = CStr16;
67
68    fn deref(&self) -> &Self::Target {
69        unsafe { CStr16::from_ptr(self.0.as_ptr().as_ptr().cast()) }
70    }
71}

97/// Protocol for converting a [`DevicePath`] or `DevicePathNode`] to a string.
98#[derive(Debug)]
99#[repr(transparent)]
100#[unsafe_protocol(DevicePathToTextProtocol::GUID)]
101pub struct DevicePathToText(DevicePathToTextProtocol);
102
103impl DevicePathToText {
104    /// Convert a [`DevicePathNode`] to a [`PoolString`].
105    ///
106    /// Returns an [`OUT_OF_RESOURCES`] error if there is insufficient
107    /// memory for the conversion.
108    ///
109    /// [`OUT_OF_RESOURCES`]: Status::OUT_OF_RESOURCES
110    pub fn convert_device_node_to_text(
111        &self,
112        device_node: &DevicePathNode,
113        display_only: DisplayOnly,
114        allow_shortcuts: AllowShortcuts,
115    ) -> Result<PoolString> {
116        let text = unsafe {
117            (self.0.convert_device_node_to_text)(
118                device_node.as_ffi_ptr().cast(),
119                display_only.0,
120                allow_shortcuts.0,
121            )
122        };
123        PoolString::new(text.cast())
124    }
125
126    /// Convert a [`DevicePath`] to a [`PoolString`].
127    ///
128    /// Returns an [`OUT_OF_RESOURCES`] error if there is insufficient
129    /// memory for the conversion.
130    ///
131    /// [`OUT_OF_RESOURCES`]: Status::OUT_OF_RESOURCES
132    pub fn convert_device_path_to_text(
133        &self,
134        device_path: &DevicePath,
135        display_only: DisplayOnly,
136        allow_shortcuts: AllowShortcuts,
137    ) -> Result<PoolString> {
138        let text = unsafe {
139            (self.0.convert_device_path_to_text)(
140                device_path.as_ffi_ptr().cast(),
141                display_only.0,
142                allow_shortcuts.0,
143            )
144        };
145        PoolString::new(text.cast())
146    }
147}

149/// Protocol for converting a string to a [`DevicePath`] or `DevicePathNode`].
150#[derive(Debug)]
151#[repr(transparent)]
152#[unsafe_protocol("05c99a21-c70f-4ad2-8a5f-35df3343f51e")]
153pub struct DevicePathFromText(DevicePathFromTextProtocol);
154
155impl DevicePathFromText {
156    /// Convert a [`CStr16`] to a [`DevicePathNode`].
157    ///
158    /// If a non-device-node character is encountered, the rest of the string is ignored.
159    ///
160    /// Returns an [`OUT_OF_RESOURCES`] error if there is insufficient
161    /// memory for the conversion.
162    ///
163    /// [`OUT_OF_RESOURCES`]: Status::OUT_OF_RESOURCES
164    pub fn convert_text_to_device_node(
165        &self,
166        text_device_node: &CStr16,
167    ) -> Result<PoolDevicePathNode> {
168        unsafe {
169            let ptr = (self.0.convert_text_to_device_node)(text_device_node.as_ptr().cast());
170            NonNull::new(ptr.cast_mut())
171                .map(|p| PoolDevicePathNode(PoolAllocation::new(p.cast())))
172                .ok_or(Status::OUT_OF_RESOURCES.into())
173        }
174    }
175
176    /// Convert a [`CStr16`] to a [`DevicePath`].
177    ///
178    /// If a non-device-node character is encountered, the rest of the string is ignored.
179    ///
180    /// Returns an [`OUT_OF_RESOURCES`] error if there is insufficient
181    /// memory for the conversion.
182    ///
183    /// [`OUT_OF_RESOURCES`]: Status::OUT_OF_RESOURCES
184    pub fn convert_text_to_device_path(&self, text_device_path: &CStr16) -> Result<PoolDevicePath> {
185        unsafe {
186            let ptr = (self.0.convert_text_to_device_path)(text_device_path.as_ptr().cast());
187            NonNull::new(ptr.cast_mut())
188                .map(|p| PoolDevicePath(PoolAllocation::new(p.cast())))
189                .ok_or(Status::OUT_OF_RESOURCES.into())
190        }
191    }
192}
