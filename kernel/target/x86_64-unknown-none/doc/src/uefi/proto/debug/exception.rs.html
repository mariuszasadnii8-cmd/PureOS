3/// Represents supported CPU exceptions.
4#[repr(C)]
5#[derive(Debug)]
6pub struct ExceptionType(isize);
7
8impl ExceptionType {
9    /// Undefined Exception
10    pub const EXCEPT_EBC_UNDEFINED: Self = Self(0);
11    /// Divide-by-zero Error
12    pub const EXCEPT_EBC_DIVIDE_ERROR: Self = Self(1);
13    /// Debug Exception
14    pub const EXCEPT_EBC_DEBUG: Self = Self(2);
15    /// Breakpoint
16    pub const EXCEPT_EBC_BREAKPOINT: Self = Self(3);
17    /// Overflow
18    pub const EXCEPT_EBC_OVERFLOW: Self = Self(4);
19    /// Invalid Opcode
20    pub const EXCEPT_EBC_INVALID_OPCODE: Self = Self(5);
21    /// Stack-Segment Fault
22    pub const EXCEPT_EBC_STACK_FAULT: Self = Self(6);
23    /// Alignment Check
24    pub const EXCEPT_EBC_ALIGNMENT_CHECK: Self = Self(7);
25    /// Instruction Encoding Exception
26    pub const EXCEPT_EBC_INSTRUCTION_ENCODING: Self = Self(8);
27    /// Bad Breakpoint Exception
28    pub const EXCEPT_EBC_BAD_BREAK: Self = Self(9);
29    /// Single Step Exception
30    pub const EXCEPT_EBC_SINGLE_STEP: Self = Self(10);
31}
