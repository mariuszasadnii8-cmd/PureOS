3//! Multi-processor management protocols.
4//!
5//! On any system with more than one logical processor we can categorize them as:
6//!
7//! * BSP — bootstrap processor, executes modules that are necessary for booting the system
8//! * AP — application processor, any processor other than the bootstrap processor
9//!
10//! This module contains protocols that provide a generalized way of performing the following tasks on these logical processors:
11//!
12//! * retrieving information of multi-processor environment and MP-related status of specific processors
13//! * dispatching user-provided function to APs
14//! * maintaining MP-related processor status
15
16use crate::data_types::Event;
17use crate::proto::unsafe_protocol;
18use crate::{Result, Status, StatusExt};
19use bitflags::bitflags;

22use core::time::Duration;

24/// Callback to be called on the AP.
25pub type Procedure = extern "efiapi" fn(*mut c_void);
26
27bitflags! {
28    /// Flags indicating if the processor is BSP or AP,
29    /// if the processor is enabled or disabled, and if
30    /// the processor is healthy.
31    #[repr(transparent)]
32    #[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord)]
33    struct StatusFlag: u32 {
34        /// Processor is playing the role of BSP.
35        const PROCESSOR_AS_BSP_BIT = 1;
36        /// Processor is enabled.
37        const PROCESSOR_ENABLED_BIT = 1 << 1;
38        /// Processor is healthy.
39        const PROCESSOR_HEALTH_STATUS_BIT = 1 << 2;
40    }
41}
42
43/// Information about number of logical processors on the platform.
44#[derive(Default, Debug)]
45pub struct ProcessorCount {
46    /// Total number of processors (including BSP).
47    pub total: usize,
48    /// Number of processors (including BSP) that are currently enabled.
49    pub enabled: usize,
50}
51
52/// Information about processor on the platform.
53#[repr(C)]
54#[derive(Default, Debug)]
55pub struct ProcessorInformation {
56    /// Unique processor ID determined by system hardware.
57    pub processor_id: u64,
58    /// Flags indicating BSP, enabled and healthy status.
59    status_flag: StatusFlag,
60    /// Physical location of the processor.
61    pub location: CpuPhysicalLocation,
62}
63
64impl ProcessorInformation {
65    /// Returns `true` if the processor is playing the role of BSP.
66    #[must_use]
67    pub const fn is_bsp(&self) -> bool {
68        self.status_flag.contains(StatusFlag::PROCESSOR_AS_BSP_BIT)
69    }
70
71    /// Returns `true` if the processor is enabled.
72    #[must_use]
73    pub const fn is_enabled(&self) -> bool {
74        self.status_flag.contains(StatusFlag::PROCESSOR_ENABLED_BIT)
75    }
76
77    /// Returns `true` if the processor is healthy.
78    #[must_use]
79    pub const fn is_healthy(&self) -> bool {
80        self.status_flag
81            .contains(StatusFlag::PROCESSOR_HEALTH_STATUS_BIT)
82    }
83}

85/// Information about physical location of the processor.
86#[repr(C)]
87#[derive(Default, Debug)]
88pub struct CpuPhysicalLocation {
89    /// Zero-based physical package number that identifies
90    /// the cartridge of the processor.
91    pub package: u32,
92    /// Zero-based physical core number within package of the processor.
93    pub core: u32,
94    /// Zero-based logical thread number within core of the processor.
95    pub thread: u32,
96}

98/// Protocol that provides services needed for multi-processor management.
99#[derive(Debug)]
100#[repr(C)]
101#[unsafe_protocol("3fdda605-a76e-4f46-ad29-12f4531b3d08")]
102pub struct MpServices {
103    get_number_of_processors: extern "efiapi" fn(
104        this: *const MpServices,
105        number_of_processors: *mut usize,
106        number_of_enabled_processors: *mut usize,
107    ) -> Status,
108    get_processor_info: extern "efiapi" fn(
109        this: *const MpServices,
110        processor_number: usize,
111        processor_info_buffer: *mut ProcessorInformation,
112    ) -> Status,
113    startup_all_aps: extern "efiapi" fn(
114        this: *const MpServices,
115        procedure: Procedure,
116        single_thread: bool,
117        wait_event: *mut c_void,
118        timeout_in_micro_seconds: usize,
119        procedure_argument: *mut c_void,
120        failed_cpu_list: *mut *mut usize,
121    ) -> Status,
122    startup_this_ap: extern "efiapi" fn(
123        this: *const MpServices,
124        procedure: Procedure,
125        processor_number: usize,
126        wait_event: *mut c_void,
127        timeout_in_micro_seconds: usize,
128        procedure_argument: *mut c_void,
129        finished: *mut bool,
130    ) -> Status,
131    switch_bsp: extern "efiapi" fn(
132        this: *const MpServices,
133        processor_number: usize,
134        enable_old_bsp: bool,
135    ) -> Status,
136    enable_disable_ap: extern "efiapi" fn(
137        this: *const MpServices,
138        processor_number: usize,
139        enable_ap: bool,
140        health_flag: *const u32,
141    ) -> Status,
142    who_am_i: extern "efiapi" fn(this: *const MpServices, processor_number: *mut usize) -> Status,
143}
144
145impl MpServices {
146    /// Retrieves the number of logical processors and the number of enabled logical processors in the system.
147    pub fn get_number_of_processors(&self) -> Result<ProcessorCount> {
148        let mut total: usize = 0;
149        let mut enabled: usize = 0;
150        (self.get_number_of_processors)(self, &mut total, &mut enabled)
151            .to_result_with_val(|| ProcessorCount { total, enabled })
152    }
153
154    /// Gets detailed information on the requested processor at the instant this call is made.
155    pub fn get_processor_info(&self, processor_number: usize) -> Result<ProcessorInformation> {
156        let mut pi: ProcessorInformation = Default::default();
157        (self.get_processor_info)(self, processor_number, &mut pi).to_result_with_val(|| pi)
158    }
159
160    /// Executes provided function on all APs.
161    pub fn startup_all_aps(
162        &self,
163        single_thread: bool,
164        procedure: Procedure,
165        procedure_argument: *mut c_void,
166        event: Option<Event>,
167        timeout: Option<Duration>,
168    ) -> Result {
169        let timeout_arg = match timeout {
170            Some(timeout) => timeout.as_micros().try_into().unwrap(),
171            None => 0,
172        };
173
174        let event_arg = match event {
175            Some(event) => event.as_ptr(),
176            None => ptr::null_mut(),
177        };
178
179        (self.startup_all_aps)(
180            self,
181            procedure,
182            single_thread,
183            event_arg,
184            timeout_arg,
185            procedure_argument,
186            ptr::null_mut(),
187        )
188        .to_result()
189    }
190
191    /// Executes provided function on a specific AP in blocking mode.
192    pub fn startup_this_ap(
193        &self,
194        processor_number: usize,
195        procedure: Procedure,
196        procedure_argument: *mut c_void,
197        event: Option<Event>,
198        timeout: Option<Duration>,
199    ) -> Result {
200        let timeout_arg = match timeout {
201            Some(timeout) => timeout.as_micros().try_into().unwrap(),
202            None => 0,
203        };
204
205        let event_arg = match event {
206            Some(event) => event.as_ptr(),
207            None => ptr::null_mut(),
208        };
209
210        (self.startup_this_ap)(
211            self,
212            procedure,
213            processor_number,
214            event_arg,
215            timeout_arg,
216            procedure_argument,
217            ptr::null_mut(),
218        )
219        .to_result()
220    }
221
222    /// Switches the requested AP to be the BSP from that point onward.
223    pub fn switch_bsp(&self, processor_number: usize, enable_old_bsp: bool) -> Result {
224        (self.switch_bsp)(self, processor_number, enable_old_bsp).to_result()
225    }
226
227    /// Enables or disables an AP from this point onward.
228    ///
229    /// The `healthy` argument can be used to specify the new health status of the AP.
230    pub fn enable_disable_ap(
231        &self,
232        processor_number: usize,
233        enable_ap: bool,
234        healthy: Option<bool>,
235    ) -> Result {
236        let health_flag_raw: u32;
237        let health_flag_ptr = match healthy {
238            Some(healthy) => {
239                let mut sf = StatusFlag::empty();
240                sf.set(StatusFlag::PROCESSOR_HEALTH_STATUS_BIT, healthy);
241                health_flag_raw = sf.bits();
242                &health_flag_raw
243            }
244            None => ptr::null(),
245        };
246        (self.enable_disable_ap)(self, processor_number, enable_ap, health_flag_ptr).to_result()
247    }
248
249    /// Gets the handle number of the caller processor.
250    pub fn who_am_i(&self) -> Result<usize> {
251        let mut processor_number: usize = 0;
252        (self.who_am_i)(self, &mut processor_number).to_result_with_val(|| processor_number)
253    }
