3//! Pointer device access.
4
5use crate::proto::unsafe_protocol;
6use crate::{Event, Result, Status, StatusExt};
7use uefi_raw::protocol::console::SimplePointerProtocol;
8
9/// Provides information about a pointer device.
10#[derive(Debug)]
11#[repr(transparent)]
12#[unsafe_protocol(SimplePointerProtocol::GUID)]
13pub struct Pointer(SimplePointerProtocol);
14
15impl Pointer {
16    /// Resets the pointer device hardware.
17    ///
18    /// The `extended_verification` parameter is used to request that UEFI
19    /// performs an extended check and reset of the input device.
20    ///
21    /// # Errors
22    ///
23    /// - `DeviceError` if the device is malfunctioning and cannot be reset.
24    pub fn reset(&mut self, extended_verification: bool) -> Result {
25        unsafe { (self.0.reset)(&mut self.0, extended_verification) }.to_result()
26    }
27
28    /// Retrieves the pointer device's current state, if a state change occurred
29    /// since the last time this function was called.
30    ///
31    /// Use `wait_for_input_event()` with the [`boot::wait_for_event`]
32    /// interface in order to wait for input from the pointer device.
33    ///
34    /// # Errors
35    /// - `DeviceError` if there was an issue with the pointer device.
36    ///
37    /// [`boot::wait_for_event`]: crate::boot::wait_for_event
38    pub fn read_state(&mut self) -> Result<Option<PointerState>> {
39        let mut pointer_state = PointerState::default();
40        let pointer_state_ptr: *mut _ = &mut pointer_state;
41
42        match unsafe { (self.0.get_state)(&mut self.0, pointer_state_ptr.cast()) } {
43            Status::NOT_READY => Ok(None),
44            other => other.to_result_with_val(|| Some(pointer_state)),
45        }
46    }
47
48    /// Event to be used with [`boot::wait_for_event`] in order to wait
49    /// for input from the pointer device
50    ///
51    /// [`boot::wait_for_event`]: crate::boot::wait_for_event
52    #[must_use]
53    pub fn wait_for_input_event(&self) -> Option<Event> {
54        unsafe { Event::from_ptr(self.0.wait_for_input) }
55    }
56
57    /// Returns a reference to the pointer device information.
58    #[must_use]
59    pub const fn mode(&self) -> &PointerMode {
60        unsafe { &*self.0.mode.cast() }
61    }
62}
63
64/// Information about this pointer device.
65#[derive(Debug, Default, Copy, Clone, Eq, PartialEq)]
66#[repr(C)]
67pub struct PointerMode {
68    /// The pointer device's resolution on the X/Y/Z axis in counts/mm.
69    /// If a value is 0, then the device does _not_ support that axis.
70    pub resolution: [u64; 3],
71    /// Whether the devices has a left button / right button.
72    pub has_button: [bool; 2],
73}
74
75/// The relative change in the pointer's state.
76#[derive(Debug, Default, Copy, Clone, Eq, PartialEq)]
77#[repr(C)]
78pub struct PointerState {
79    /// The relative movement on the X/Y/Z axis.
80    ///
81    /// If `PointerMode` indicates an axis is not supported, it must be ignored.
82    pub relative_movement: [i32; 3],
83    /// Whether the left / right mouse button is currently pressed.
84    ///
85    /// If `PointerMode` indicates a button is not supported, it must be ignored.
86    pub button: [bool; 2],
87}
