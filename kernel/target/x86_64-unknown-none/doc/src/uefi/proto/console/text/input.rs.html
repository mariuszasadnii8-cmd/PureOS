8/// Interface for text-based input devices.
9#[derive(Debug)]
10#[repr(transparent)]
11#[unsafe_protocol(SimpleTextInputProtocol::GUID)]
12pub struct Input(SimpleTextInputProtocol);
13
14impl Input {
15    /// Resets the input device hardware.
16    ///
17    /// The `extended_verification` parameter is used to request that UEFI
18    /// performs an extended check and reset of the input device.
19    ///
20    /// # Errors
21    ///
22    /// - `DeviceError` if the device is malfunctioning and cannot be reset.
23    pub fn reset(&mut self, extended_verification: bool) -> Result {
24        unsafe { (self.0.reset)(&mut self.0, extended_verification) }.to_result()
25    }
26
27    /// Reads the next keystroke from the input device, if any.
28    ///
29    /// Use [`wait_for_key_event`] with the [`boot::wait_for_event`]
30    /// interface in order to wait for a key to be pressed.
31    ///
32    /// [`boot::wait_for_event`]: crate::boot::wait_for_event
33    /// [`wait_for_key_event`]: Self::wait_for_key_event
34    ///
35    /// # Errors
36    ///
37    /// - [`Status::DEVICE_ERROR`] if there was an issue with the input device
38    ///
39    /// # Examples
40    ///
41    /// ```
42    /// use log::info;
43    /// use uefi::proto::console::text::{Input, Key, ScanCode};
44    /// use uefi::{boot, Char16, Result, ResultExt};
45    ///
46    /// fn read_keyboard_events(input: &mut Input) -> Result {
47    ///     loop {
48    ///         // Pause until a keyboard event occurs.
49    ///         let mut events = unsafe { [input.wait_for_key_event().unwrap()] };
50    ///         boot::wait_for_event(&mut events).discard_errdata()?;
51    ///
52    ///         let u_key = Char16::try_from('u').unwrap();
53    ///         match input.read_key()? {
54    ///             // Example of handling a printable key: print a message when
55    ///             // the 'u' key is pressed.
56    ///             Some(Key::Printable(key)) if key == u_key => {
57    ///                 info!("the 'u' key was pressed");
58    ///             }
59    ///
60    ///             // Example of handling a special key: exit the loop when the
61    ///             // escape key is pressed.
62    ///             Some(Key::Special(ScanCode::ESCAPE)) => {
63    ///                 break;
64    ///             }
65    ///             _ => {}
66    ///         }
67    ///     }
68    ///
69    ///     Ok(())
70    /// }
71    /// ```
72    pub fn read_key(&mut self) -> Result<Option<Key>> {
73        let mut key = MaybeUninit::<InputKey>::uninit();
74
75        match unsafe { (self.0.read_key_stroke)(&mut self.0, key.as_mut_ptr()) } {
76            Status::NOT_READY => Ok(None),
77            other => other.to_result_with_val(|| Some(unsafe { key.assume_init() }.into())),
78        }
79    }
80
81    /// Event to be used with [`boot::wait_for_event`] in order to wait
82    /// for a key to be available
83    ///
84    /// [`boot::wait_for_event`]: crate::boot::wait_for_event
85    #[must_use]
86    pub fn wait_for_key_event(&self) -> Option<Event> {
87        unsafe { Event::from_ptr(self.0.wait_for_key) }
88    }
89}

91/// A key read from the console (high-level version)
92#[derive(Debug, Copy, Clone, Eq, PartialEq)]
93pub enum Key {
94    /// The key is associated with a printable Unicode character
95    Printable(Char16),
96
97    /// The key is special (arrow, function, multimedia...)
98    Special(ScanCode),
99}
100
101impl From<InputKey> for Key {
102    fn from(k: InputKey) -> Self {
103        if k.scan_code == ScanCode::NULL.0 {
104            Self::Printable(Char16::try_from(k.unicode_char).unwrap())
105        } else {
106            Self::Special(ScanCode(k.scan_code))
107        }
108    }
109}
