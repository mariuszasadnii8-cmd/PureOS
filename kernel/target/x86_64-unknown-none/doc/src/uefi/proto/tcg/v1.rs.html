35/// Information about the protocol and the TPM device.
36///
37/// Layout compatible with the C type `TCG_EFI_BOOT_SERVICE_CAPABILITY`.
38#[repr(C)]
39#[derive(Clone, Copy, Debug, Default, Eq, PartialEq, Ord, PartialOrd)]
40pub struct BootServiceCapability(TcgBootServiceCapability);
41
42impl BootServiceCapability {
43    /// Version of the `BootServiceCapability` structure.
44    #[must_use]
45    pub const fn structure_version(&self) -> Version {
46        self.0.structure_version
47    }
48
49    /// Version of the `Tcg` protocol.
50    #[must_use]
51    pub const fn protocol_spec_version(&self) -> Version {
52        self.0.protocol_spec_version
53    }
54
55    /// Supported hash algorithms.
56    #[must_use]
57    pub fn hash_algorithm(&self) -> HashAlgorithm {
58        // Safety: the value should always be 0x1 (indicating SHA-1), but
59        // we don't care if it's some unexpected value.
60        HashAlgorithm::from_bits_retain(u32::from(self.0.hash_algorithm_bitmap))
61    }
62
63    /// Whether the TPM device is present.
64    #[must_use]
65    pub const fn tpm_present(&self) -> bool {
66        self.0.tpm_present_flag != 0
67    }
68
69    /// Whether the TPM device is deactivated.
70    #[must_use]
71    pub const fn tpm_deactivated(&self) -> bool {
72        self.0.tpm_deactivated_flag != 0
73    }
74}

76/// Entry in the [`EventLog`].
77///
78/// Layout compatible with the C type `TCG_PCR_EVENT`.
79///
80/// Naming note: the spec refers to "event data" in two conflicting
81/// ways: the `event_data` field and the data hashed in the digest
82/// field. These two are independent; although the event data _can_ be
83/// what is hashed in the digest field, it doesn't have to be.
84#[repr(C, packed)]
85#[derive(Eq, Pointee)]
86pub struct PcrEvent {
87    pcr_index: PcrIndex,
88    event_type: EventType,
89    digest: Sha1Digest,
90    event_data_size: u32,
91    event_data: [u8],
92}
93
94impl PcrEvent {
95    pub(super) const unsafe fn from_ptr<'a>(ptr: *const u8) -> &'a Self {
96        // Get the `event_size` field.
97        let ptr_u32: *const u32 = ptr.cast();
98        let event_size = ptr_u32.add(7).read_unaligned();
99        let event_size = usize_from_u32(event_size);
100        unsafe { &*ptr_meta::from_raw_parts(ptr.cast(), event_size) }
101    }
102
103    /// Create a new `PcrEvent` using a byte buffer for storage.
104    ///
105    /// # Errors
106    ///
107    /// Returns [`Status::BUFFER_TOO_SMALL`] if the `buffer` is not large
108    /// enough. The required size will be returned in the error data.
109    ///
110    /// Returns [`Status::INVALID_PARAMETER`] if the `event_data` size is too
111    /// large.
112    pub fn new_in_buffer<'buf>(
113        buffer: &'buf mut [u8],
114        pcr_index: PcrIndex,
115        event_type: EventType,
116        digest: Sha1Digest,
117        event_data: &[u8],
118    ) -> Result<&'buf mut Self, Option<usize>> {
119        let event_data_size = u32::try_from(event_data.len())
120            .map_err(|_| Error::new(Status::INVALID_PARAMETER, None))?;
121
122        let required_size = size_of::<PcrIndex>()
123            + size_of::<EventType>()
124            + size_of::<Sha1Digest>()
125            + size_of::<u32>()
126            + event_data.len();
127
128        if buffer.len() < required_size {
129            return Err(Error::new(Status::BUFFER_TOO_SMALL, Some(required_size)));
130        }
131
132        let mut ptr: *mut u8 = buffer.as_mut_ptr().cast();
133
134        unsafe {
135            ptr_write_unaligned_and_add(&mut ptr, pcr_index);
136            ptr_write_unaligned_and_add(&mut ptr, event_type);
137            ptr_write_unaligned_and_add(&mut ptr, digest);
138            ptr_write_unaligned_and_add(&mut ptr, event_data_size);
139            ptr::copy(event_data.as_ptr(), ptr, event_data.len());
140
141            let ptr: *mut Self =
142                ptr_meta::from_raw_parts_mut(buffer.as_mut_ptr().cast(), event_data.len());
143            Ok(&mut *ptr)
144        }
145    }
146
147    /// Create a new `PcrEvent` in a [`Box`].
148    ///
149    /// # Errors
150    ///
151    /// Returns [`Status::INVALID_PARAMETER`] if the `event_data` size is too
152    /// large.
153    #[cfg(feature = "alloc")]
154    pub fn new_in_box(
155        pcr_index: PcrIndex,
156        event_type: EventType,
157        digest: Sha1Digest,
158        event_data: &[u8],
159    ) -> Result<Box<Self>> {
160        #[cfg(not(feature = "unstable"))]
161        {
162            make_boxed(|buf| Self::new_in_buffer(buf, pcr_index, event_type, digest, event_data))
163        }
164        #[cfg(feature = "unstable")]
165        {
166            make_boxed(
167                |buf| Self::new_in_buffer(buf, pcr_index, event_type, digest, event_data),
168                Global,
169            )
170        }
171    }
172
173    /// PCR index for the event.
174    #[must_use]
175    pub const fn pcr_index(&self) -> PcrIndex {
176        self.pcr_index
177    }
178
179    /// Type of event, indicating what type of data is stored in [`event_data`].
180    ///
181    /// [`event_data`]: Self::event_data
182    #[must_use]
183    pub const fn event_type(&self) -> EventType {
184        self.event_type
185    }
186
187    /// Raw event data. The meaning of this data can be determined from
188    /// the [`event_type`].
189    ///
190    /// Note that this data is independent of what is hashed [`digest`].
191    ///
192    /// [`digest`]: Self::digest
193    /// [`event_type`]: Self::event_type
194    #[must_use]
195    pub const fn event_data(&self) -> &[u8] {
196        &self.event_data
197    }
198
199    /// SHA-1 digest of the data hashed for this event.
200    #[must_use]
201    pub const fn digest(&self) -> Sha1Digest {
202        self.digest
203    }

212// Manual `Debug` implementation since it can't be derived for a packed DST.
213impl Debug for PcrEvent {
214    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
215        f.debug_struct("PcrEvent")
216            .field("pcr_index", &{ self.pcr_index })
217            .field("event_type", &{ self.event_type })
218            .field("digest", &self.digest)
219            .field("event_data_size", &{ self.event_data_size })
220            .field("event_data", &&self.event_data)
221            .finish()
222    }
223}

225// Manual `PartialEq` implementation since it can't be derived for a packed DST.
226impl PartialEq for PcrEvent {
227    fn eq(&self, rhs: &Self) -> bool {
228        self.pcr_index() == rhs.pcr_index()
229            && self.event_type() == rhs.event_type()
230            && self.digest == rhs.digest
231            && self.event_data_size == rhs.event_data_size
232            && self.event_data == rhs.event_data
233    }

243/// TPM event log.
244///
245/// This type of event log always uses SHA-1 hashes. The [`v1::Tcg`]
246/// protocol always uses this type of event log, but it can also be
247/// provided by the [`v2::Tcg`] protocol via [`get_event_log_v2`].
248///
249/// [`v1::Tcg`]: Tcg
250/// [`v2::Tcg`]: super::v2::Tcg
251/// [`get_event_log_v2`]: super::v2::Tcg::get_event_log_v2
252#[derive(Debug)]
253pub struct EventLog<'a> {
254    // Tie the lifetime to the protocol, and by extension, boot services.
255    _lifetime: PhantomData<&'a Tcg>,
256
257    location: *const u8,
258    last_entry: *const u8,
259
260    is_truncated: bool,
261}
262
263impl EventLog<'_> {
264    pub(super) const unsafe fn new(
265        location: *const u8,
266        last_entry: *const u8,
267        is_truncated: bool,
268    ) -> Self {
269        Self {
270            _lifetime: PhantomData,
271            location,
272            last_entry,
273            is_truncated,
274        }
275    }
276
277    /// Iterator of events in the log.
278    #[must_use]
279    pub const fn iter(&self) -> EventLogIter {
280        EventLogIter {
281            log: self,
282            location: self.location,
283        }
284    }
285
286    /// If true, the event log is missing one or more entries because
287    /// additional events would have exceeded the space allocated for
288    /// the log.
289    ///
290    /// This value is not reported for the [`v1::Tcg`] protocol, so it
291    /// is always `false` in that case.
292    ///
293    /// [`v1::Tcg`]: Tcg
294    #[must_use]
295    pub const fn is_truncated(&self) -> bool {
296        self.is_truncated
297    }
298}

300/// Iterator for events in [`EventLog`].
301#[derive(Debug)]
302pub struct EventLogIter<'a> {
303    log: &'a EventLog<'a>,
304    location: *const u8,
305}
306
307impl<'a> Iterator for EventLogIter<'a> {
308    type Item = &'a PcrEvent;
309
310    fn next(&mut self) -> Option<Self::Item> {
311        // The spec says that `last_entry` will be null if there are no
312        // events. Presumably `location` will be null as well, but check
313        // both just to be safe.
314        if self.location.is_null() || self.log.last_entry.is_null() {
315            return None;
316        }
317
318        // Safety: we trust that the protocol has given us a valid range
319        // of memory to read from.
320        let event = unsafe { PcrEvent::from_ptr(self.location) };
321
322        // If this is the last entry, set the location to null so that
323        // future calls to `next()` return `None`.
324        if self.location == self.log.last_entry {
325            self.location = ptr::null();
326        } else {
327            self.location = unsafe { self.location.add(size_of_val(event)) };
328        }
329
330        Some(event)
331    }
332}

342/// Return type of [`Tcg::status_check`].
343#[derive(Debug)]
344pub struct StatusCheck<'a> {
345    /// Information about the protocol and the TPM device.
346    pub protocol_capability: BootServiceCapability,
347
348    /// Feature flags. The spec does not define any feature flags, so
349    /// this is always expected to be zero.
350    pub feature_flags: u32,
351
352    /// TPM event log.
353    pub event_log: EventLog<'a>,
354}
355
356impl Tcg {
357    /// Get information about the protocol and TPM device, as well as
358    /// the TPM event log.
359    pub fn status_check(&mut self) -> Result<StatusCheck> {
360        let mut protocol_capability = TcgBootServiceCapability::default();
361        let mut feature_flags = 0;
362        let mut event_log_location = 0;
363        let mut event_log_last_entry = 0;
364
365        let status = unsafe {
366            (self.0.status_check)(
367                &mut self.0,
368                &mut protocol_capability,
369                &mut feature_flags,
370                &mut event_log_location,
371                &mut event_log_last_entry,
372            )
373        };
374
375        if status.is_success() {
376            // The truncated field is just there for the v2 protocol;
377            // always set it to false for v1.
378            let truncated = false;
379            let event_log = unsafe {
380                EventLog::new(
381                    event_log_location as *const u8,
382                    event_log_last_entry as *const u8,
383                    truncated,
384                )
385            };
386
387            Ok(StatusCheck {
388                protocol_capability: BootServiceCapability(protocol_capability),
389                feature_flags,
390                event_log,
391            })
392        } else {
393            Err(status.into())
394        }
395    }
396
397    /// Add an entry to the event log without extending a PCR.
398    ///
399    /// Usually [`hash_log_extend_event`] should be used instead. An
400    /// entry added via `log_event` cannot be verified, so it is mainly
401    /// intended for adding an informational entry.
402    ///
403    /// [`hash_log_extend_event`]: Self::hash_log_extend_event
404    pub fn log_event(&mut self, event: &PcrEvent) -> Result {
405        // This is the only valid value; it indicates that the extend
406        // operation should not be performed.
407        let flags = 0x1;
408
409        // Don't bother returning this, it's not very useful info.
410        let mut event_number = 0;
411
412        let event_ptr: *const PcrEvent = event;
413
414        unsafe {
415            (self.0.log_event)(&mut self.0, event_ptr.cast(), &mut event_number, flags).to_result()
416        }
417    }
418
419    /// Extend a PCR and add an entry to the event log.
420    ///
421    /// If `data_to_hash` is `None` then the `digest` field of the `event`
422    /// should be used as-is. Otherwise, the `digest` field will be overwritten
423    /// with the SHA-1 hash of the data.
424    pub fn hash_log_extend_event(
425        &mut self,
426        event: &mut PcrEvent,
427        data_to_hash: Option<&[u8]>,
428    ) -> Result {
429        let hash_data;
430        let hash_data_len;
431        if let Some(data_to_hash) = data_to_hash {
432            hash_data = data_to_hash.as_ptr() as PhysicalAddress;
433            hash_data_len = u64::try_from(data_to_hash.len()).unwrap();
434        } else {
435            hash_data = 0;
436            hash_data_len = 0;
437        }
438
439        // Don't bother returning these, it's not very useful info.
440        let mut event_number = 0;
441        let mut event_log_last_entry = 0;
442
443        let event_ptr: *mut PcrEvent = event;
444
445        unsafe {
446            (self.0.hash_log_extend_event)(
447                &mut self.0,
448                hash_data,
449                hash_data_len,
450                AlgorithmId::SHA1.0.into(),
451                event_ptr.cast(),
452                &mut event_number,
453                &mut event_log_last_entry,
454            )
455            .to_result()
456        }
457    }
458
459    /// Send a command directly to the TPM.
460    ///
461    /// Constructing the input block and parsing the output block are outside
462    /// the scope of this crate. See the [TPM 1.2 Main Specification][spec]
463    /// documents for details of these blocks, in particular Part 3, Commands.
464    ///
465    /// Note that TPM structures are big endian.
466    ///
467    /// [spec]: https://trustedcomputinggroup.org/resource/tpm-main-specification/
468    pub fn pass_through_to_tpm(
469        &mut self,
470        input_parameter_block: &[u8],
471        output_parameter_block: &mut [u8],
472    ) -> Result {
473        let input_parameter_block_len = u32::try_from(input_parameter_block.len())
474            .map_err(|_| Error::from(Status::BAD_BUFFER_SIZE))?;
475        let output_parameter_block_len = u32::try_from(output_parameter_block.len())
476            .map_err(|_| Error::from(Status::BAD_BUFFER_SIZE))?;
477
478        unsafe {
479            (self.0.pass_through_to_tpm)(
480                &mut self.0,
481                input_parameter_block_len,
482                input_parameter_block.as_ptr(),
483                output_parameter_block_len,
484                output_parameter_block.as_mut_ptr(),
485            )
486            .to_result()
487        }
488    }

496    #[test]
497    fn test_new_pcr_event() {
498        let mut event_buf = [0; 256];
499        #[rustfmt::skip]
500        let digest = [
501            0x00, 0x01, 0x02, 0x03,
502            0x04, 0x05, 0x06, 0x07,
503            0x08, 0x09, 0x0a, 0x0b,
504            0x0c, 0x0d, 0x0e, 0x0f,
505            0x10, 0x11, 0x12, 0x13,
506        ];
507        let data = [0x14, 0x15, 0x16, 0x17];
508        let event =
509            PcrEvent::new_in_buffer(&mut event_buf, PcrIndex(4), EventType::IPL, digest, &data)
510                .unwrap();
511        assert_eq!(event.pcr_index(), PcrIndex(4));
512        assert_eq!(event.event_type(), EventType::IPL);
513        assert_eq!(event.digest(), digest);
514        assert_eq!(event.event_data(), data);
515
516        let event_ptr: *const PcrEvent = event;
517        let bytes = unsafe { slice::from_raw_parts(event_ptr.cast::<u8>(), size_of_val(event)) };
518        #[rustfmt::skip]
519        assert_eq!(bytes, [
520            // PCR index
521            0x04, 0x00, 0x00, 0x00,
522            // Event type
523            0x0d, 0x00, 0x00, 0x00,
524            // Digest
525            0x00, 0x01, 0x02, 0x03,
526            0x04, 0x05, 0x06, 0x07,
527            0x08, 0x09, 0x0a, 0x0b,
528            0x0c, 0x0d, 0x0e, 0x0f,
529            0x10, 0x11, 0x12, 0x13,
530            // Event data len
531            0x04, 0x00, 0x00, 0x00,
532            // Event data
533            0x14, 0x15, 0x16, 0x17,
534        ]);
535
536        // Check that `new_in_box` gives the same value.
537        assert_eq!(
538            event,
539            &*PcrEvent::new_in_box(PcrIndex(4), EventType::IPL, digest, &data).unwrap()
540        );
541    }
