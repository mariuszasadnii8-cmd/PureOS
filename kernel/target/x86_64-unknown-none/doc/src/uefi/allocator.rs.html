3//! This module implements Rust's global allocator interface using UEFI's memory allocation functions.
4//!
5//! If the `global_allocator` feature is enabled, the [`Allocator`] will be used
6//! as the global Rust allocator.
7//!
8//! This allocator can only be used while boot services are active. If boot
9//! services are not active, `alloc` will return a null pointer, and `dealloc`
10//! will panic.
11
12use crate::boot;
13use crate::mem::memory_map::MemoryType;
14use crate::proto::loaded_image::LoadedImage;
15use core::alloc::{GlobalAlloc, Layout};
16use core::ptr::{self, NonNull};
17use core::sync::atomic::{AtomicU32, Ordering};
18
19/// Get the memory type to use for allocation.
20///
21/// The first time this is called, the data type of the loaded image will be
22/// retrieved. That value is cached in a static and reused on subsequent
23/// calls. If the memory type of the loaded image cannot be retrieved for some
24/// reason, a default of `LOADER_DATA` is used.
25fn get_memory_type() -> MemoryType {
26    // Initialize to a `RESERVED` to indicate the actual value hasn't been set yet.
27    static MEMORY_TYPE: AtomicU32 = AtomicU32::new(MemoryType::RESERVED.0);
28
29    let memory_type = MEMORY_TYPE.load(Ordering::Acquire);
30    if memory_type == MemoryType::RESERVED.0 {
31        let memory_type = if let Ok(loaded_image) =
32            boot::open_protocol_exclusive::<LoadedImage>(boot::image_handle())
33        {
34            loaded_image.data_type()
35        } else {
36            MemoryType::LOADER_DATA
37        };
38        MEMORY_TYPE.store(memory_type.0, Ordering::Release);
39        memory_type
40    } else {
41        MemoryType(memory_type)
42    }
43}
44
45/// Allocator which uses the UEFI pool allocation functions.
46///
47/// Only valid for as long as the UEFI boot services are available.
48#[derive(Debug)]
49pub struct Allocator;
50
51unsafe impl GlobalAlloc for Allocator {
52    /// Allocate memory using [`boot::allocate_pool`]. The allocation's [memory
53    /// type] matches the current image's [data type].
54    ///
55    /// [memory type]: MemoryType
56    /// [data type]: LoadedImage::data_type
57    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
58        if !boot::are_boot_services_active() {
59            return ptr::null_mut();
60        }
61
62        let size = layout.size();
63        let align = layout.align();
64        let memory_type = get_memory_type();
65
66        if align > 8 {
67            // The requested alignment is greater than 8, but `allocate_pool` is
68            // only guaranteed to provide eight-byte alignment. Allocate extra
69            // space so that we can return an appropriately-aligned pointer
70            // within the allocation.
71            let full_alloc_ptr = if let Ok(ptr) = boot::allocate_pool(memory_type, size + align) {
72                ptr.as_ptr()
73            } else {
74                return ptr::null_mut();
75            };
76
77            // Calculate the offset needed to get an aligned pointer within the
78            // full allocation. If that offset is zero, increase it to `align`
79            // so that we still have space to store the extra pointer described
80            // below.
81            let mut offset = full_alloc_ptr.align_offset(align);
82            if offset == 0 {
83                offset = align;
84            }
85
86            // Before returning the aligned allocation, store a pointer to the
87            // full unaligned allocation in the bytes just before the aligned
88            // allocation. We know we have at least eight bytes there due to
89            // adding `align` to the memory allocation size. We also know the
90            // write is appropriately aligned for a `*mut u8` pointer because
91            // `align_ptr` is aligned, and alignments are always powers of two
92            // (as enforced by the `Layout` type).
93            let aligned_ptr = full_alloc_ptr.add(offset);
94            (aligned_ptr.cast::<*mut u8>()).sub(1).write(full_alloc_ptr);
95            aligned_ptr
96        } else {
97            // The requested alignment is less than or equal to eight, and
98            // `allocate_pool` always provides eight-byte alignment, so we can
99            // use `allocate_pool` directly.
100            boot::allocate_pool(memory_type, size)
101                .map(|ptr| ptr.as_ptr())
102                .unwrap_or(ptr::null_mut())
103        }
104    }
105
106    /// Deallocate memory using [`boot::free_pool`].
107    unsafe fn dealloc(&self, mut ptr: *mut u8, layout: Layout) {
108        if layout.align() > 8 {
109            // Retrieve the pointer to the full allocation that was packed right
110            // before the aligned allocation in `alloc`.
111            ptr = (ptr as *const *mut u8).sub(1).read();
112        }
113
114        // OK to unwrap: `ptr` is required to be a valid allocation by the trait API.
115        let ptr = NonNull::new(ptr).unwrap();
116
117        // Warning: this will panic after exiting boot services.
118        boot::free_pool(ptr).unwrap();
119    }
120}
