10/// Opaque handle to an UEFI entity (protocol, image...), guaranteed to be non-null.
11///
12/// If you need to have a nullable handle (for a custom UEFI FFI for example) use `Option<Handle>`.
13#[derive(Clone, Copy, Debug, Hash, Eq, PartialEq, Ord, PartialOrd)]
14#[repr(transparent)]
15pub struct Handle(NonNull<c_void>);
16
17impl Handle {
18    /// Creates a new [`Handle`].
19    ///
20    /// # Safety
21    /// This function is unsafe because the caller must be sure that the pointer
22    /// is valid. Otherwise, further operations on the object might result in
23    /// undefined behaviour, even if the methods aren't marked as unsafe.
24    #[must_use]
25    pub const unsafe fn new(ptr: NonNull<c_void>) -> Self {
26        Self(ptr)
27    }
28
29    /// Creates a new [`Handle`] from a raw address. The address might
30    /// come from the Multiboot2 information structure or something similar.
31    ///
32    /// # Example
33    /// ```no_run
34    /// use core::ffi::c_void;
35    /// use uefi::Handle;
36    ///
37    /// let image_handle_addr = 0xdeadbeef as *mut c_void;
38    ///
39    /// let uefi_image_handle = unsafe {
40    ///     Handle::from_ptr(image_handle_addr).expect("Pointer must not be null!")
41    /// };
42    /// ```
43    ///
44    /// # Safety
45    /// This function is unsafe because the caller must be sure that the pointer
46    /// is valid. Otherwise, further operations on the object might result in
47    /// undefined behaviour, even if the methods aren't marked as unsafe.
48    pub unsafe fn from_ptr(ptr: *mut c_void) -> Option<Self> {
49        // shorthand for "|ptr| Self(ptr)"
50        NonNull::new(ptr).map(Self)
51    }
52
53    /// Get the underlying raw pointer.
54    #[must_use]
55    pub const fn as_ptr(&self) -> *mut c_void {
56        self.0.as_ptr()
57    }
58
59    pub(crate) fn opt_to_ptr(handle: Option<Self>) -> *mut c_void {
60        handle.map(|h| h.0.as_ptr()).unwrap_or(ptr::null_mut())
61    }
62}

64/// Handle to an event structure, guaranteed to be non-null.
65///
66/// If you need to have a nullable event, use `Option<Event>`.
67#[derive(Debug, Eq, PartialEq, Hash, Ord, PartialOrd)]
68#[repr(transparent)]
69pub struct Event(NonNull<c_void>);
70
71impl Event {
72    /// Clone this `Event`
73    ///
74    /// # Safety
75    /// When an event is closed by calling [`boot::close_event`], that event and ALL references
76    /// to it are invalidated and the underlying memory is freed by firmware. The caller must ensure
77    /// that any clones of a closed `Event` are never used again.
78    ///
79    /// [`boot::close_event`]: crate::boot::close_event
80    #[must_use]
81    pub const unsafe fn unsafe_clone(&self) -> Self {
82        Self(self.0)
83    }
84
85    /// Create an `Event` from a raw pointer.
86    ///
87    /// # Safety
88    ///
89    /// The caller must ensure that the pointer is valid.
90    pub unsafe fn from_ptr(ptr: *mut c_void) -> Option<Self> {
91        NonNull::new(ptr).map(Self)
92    }
93
94    /// Get the underlying raw pointer.
95    #[must_use]
96    pub const fn as_ptr(&self) -> *mut c_void {
97        self.0.as_ptr()
98    }
99}

101/// Trait for querying the alignment of a struct.
102///
103/// For a statically-sized type the alignment can be retrieved with
104/// [`align_of`]. For a dynamically-sized type (DST),
105/// [`align_of_val`] provides the alignment given a reference. But in
106/// some cases it's helpful to know the alignment of a DST prior to having a
107/// value, meaning there's no reference to pass to `align_of_val`. For example,
108/// when using an API that creates a value using a `[u8]` buffer, the alignment
109/// of the buffer must be checked. The `Align` trait makes that possible by
110/// allowing the appropriate alignment to be manually specified.
111pub trait Align {
112    /// Required memory alignment for this type
113    fn alignment() -> usize;
114
115    /// Calculate the offset from `val` necessary to make it aligned,
116    /// rounding up. For example, if `val` is 1 and the alignment is 8,
117    /// this will return 7. Returns 0 if `val == 0`.
118    #[must_use]
119    fn offset_up_to_alignment(val: usize) -> usize {
120        assert!(Self::alignment() != 0);
121        let r = val % Self::alignment();
122        if r == 0 {
123            0
124        } else {
125            Self::alignment() - r
126        }
127    }
128
129    /// Round `val` up so that it is aligned.
130    #[must_use]
131    fn round_up_to_alignment(val: usize) -> usize {
132        val + Self::offset_up_to_alignment(val)
133    }
134
135    /// Get a subslice of `buf` where the address of the first element
136    /// is aligned. Returns `None` if no element of the buffer is
137    /// aligned.
138    fn align_buf(buf: &mut [u8]) -> Option<&mut [u8]> {
139        let offset = buf.as_ptr().align_offset(Self::alignment());
140        buf.get_mut(offset..)
141    }
142
143    /// Assert that some storage is correctly aligned for this type
144    fn assert_aligned(storage: &mut [u8]) {
145        if !storage.is_empty() {
146            assert_eq!(
147                storage.as_ptr().align_offset(Self::alignment()),
148                0,
149                "The provided storage is not correctly aligned for this type"
150            )
151        }
152    }
153}
154
155impl Align for [u8] {
156    fn alignment() -> usize {
157        1
158    }
159}

189#[cfg(test)]
190mod tests {
191    use super::*;
192
193    #[test]
194    fn test_alignment() {
195        struct X {}
196
197        impl Align for X {
198            fn alignment() -> usize {
199                4
200            }
201        }
202
203        assert_eq!(X::offset_up_to_alignment(0), 0);
204        assert_eq!(X::offset_up_to_alignment(1), 3);
205        assert_eq!(X::offset_up_to_alignment(2), 2);
206        assert_eq!(X::offset_up_to_alignment(3), 1);
207        assert_eq!(X::offset_up_to_alignment(4), 0);
208        assert_eq!(X::offset_up_to_alignment(5), 3);
209        assert_eq!(X::offset_up_to_alignment(6), 2);
210        assert_eq!(X::offset_up_to_alignment(7), 1);
211        assert_eq!(X::offset_up_to_alignment(8), 0);
212
213        assert_eq!(X::round_up_to_alignment(0), 0);
214        assert_eq!(X::round_up_to_alignment(1), 4);
215        assert_eq!(X::round_up_to_alignment(2), 4);
216        assert_eq!(X::round_up_to_alignment(3), 4);
217        assert_eq!(X::round_up_to_alignment(4), 4);
218        assert_eq!(X::round_up_to_alignment(5), 8);
219        assert_eq!(X::round_up_to_alignment(6), 8);
220        assert_eq!(X::round_up_to_alignment(7), 8);
221        assert_eq!(X::round_up_to_alignment(8), 8);
222
223        // Get an intentionally misaligned buffer.
224        let mut buffer = [0u8; 16];
225        let mut buffer = &mut buffer[..];
226        if (buffer.as_ptr() as usize) % X::alignment() == 0 {
227            buffer = &mut buffer[1..];
228        }
229
230        let buffer = X::align_buf(buffer).unwrap();
231        X::assert_aligned(buffer);
232    }
