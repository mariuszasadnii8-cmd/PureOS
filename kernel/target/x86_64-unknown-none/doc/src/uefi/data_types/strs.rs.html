15/// Error converting from a slice (which can contain interior nuls) to a string
16/// type.

18pub enum FromSliceUntilNulError {
19    /// An invalid character was encountered before the end of the slice.

31        }
32    }
33}

119/// A null-terminated Latin-1 string.
120///
121/// This type is largely inspired by [`core::ffi::CStr`] with the exception that all characters are
122/// guaranteed to be 8 bit long.
123///
124/// A [`CStr8`] can be constructed from a [`core::ffi::CStr`] via a `try_from` call:
125/// ```ignore
126/// let cstr8: &CStr8 = TryFrom::try_from(cstr).unwrap();
127/// ```
128///
129/// For convenience, a [`CStr8`] is comparable with [`core::str`] and
130/// `alloc::string::String` from the standard library through the trait [`EqStrUntilNul`].
131#[repr(transparent)]
132#[derive(Eq, PartialEq, Ord, PartialOrd, Hash)]
133pub struct CStr8([Char8]);
134
135impl CStr8 {
136    /// Takes a raw pointer to a null-terminated Latin-1 string and wraps it in a CStr8 reference.
137    ///
138    /// # Safety
139    ///
140    /// The function will start accessing memory from `ptr` until the first
141    /// null byte. It's the callers responsibility to ensure `ptr` points to
142    /// a valid null-terminated string in accessible memory.
143    #[must_use]
144    pub unsafe fn from_ptr<'ptr>(ptr: *const Char8) -> &'ptr Self {
145        let mut len = 0;
146        while *ptr.add(len) != NUL_8 {
147            len += 1
148        }
149        let ptr = ptr.cast::<u8>();
150        Self::from_bytes_with_nul_unchecked(slice::from_raw_parts(ptr, len + 1))
151    }
152
153    /// Creates a CStr8 reference from bytes.
154    pub fn from_bytes_with_nul(chars: &[u8]) -> Result<&Self, FromSliceWithNulError> {
155        let nul_pos = chars.iter().position(|&c| c == 0);
156        if let Some(nul_pos) = nul_pos {
157            if nul_pos + 1 != chars.len() {
158                return Err(FromSliceWithNulError::InteriorNul(nul_pos));
159            }
160            Ok(unsafe { Self::from_bytes_with_nul_unchecked(chars) })
161        } else {
162            Err(FromSliceWithNulError::NotNulTerminated)
163        }
164    }
165
166    /// Unsafely creates a CStr8 reference from bytes.
167    ///
168    /// # Safety
169    ///
170    /// It's the callers responsibility to ensure chars is a valid Latin-1
171    /// null-terminated string, with no interior null bytes.
172    #[must_use]
173    pub const unsafe fn from_bytes_with_nul_unchecked(chars: &[u8]) -> &Self {
174        &*(ptr::from_ref(chars) as *const Self)
175    }
176
177    /// Returns the inner pointer to this CStr8.
178    #[must_use]
179    pub const fn as_ptr(&self) -> *const Char8 {
180        self.0.as_ptr()
181    }
182
183    /// Returns the underlying bytes as slice including the terminating null
184    /// character.
185    #[must_use]
186    pub const fn as_bytes(&self) -> &[u8] {
187        unsafe { &*(ptr::from_ref(&self.0) as *const [u8]) }
188    }
189}

193        write!(f, "CStr8({:?})", &self.0)
194    }

218impl<StrType: AsRef<str> + ?Sized> EqStrUntilNul<StrType> for CStr8 {
219    fn eq_str_until_nul(&self, other: &StrType) -> bool {
220        let other = other.as_ref();
221
222        // TODO: CStr16 has .iter() implemented, CStr8 not yet
223        let any_not_equal = self
224            .0
225            .iter()
226            .copied()
227            .map(char::from)
228            .zip(other.chars())
229            // This only works as CStr8 is guaranteed to have a fixed character length
230            // (unlike UTF-8).
231            .take_while(|(l, r)| *l != '\0' && *r != '\0')
232            .any(|(l, r)| l != r);
233
234        !any_not_equal
235    }
236}

246/// Get a Latin-1 character from a UTF-8 byte slice at the given offset.
247///
248/// Returns a pair containing the Latin-1 character and the number of bytes in
249/// the UTF-8 encoding of that character.
250///
251/// Panics if the string cannot be encoded in Latin-1.
252///
253/// # Safety
254///
255/// The input `bytes` must be valid UTF-8.
256const unsafe fn latin1_from_utf8_at_offset(bytes: &[u8], offset: usize) -> (u8, usize) {
257    if bytes[offset] & 0b1000_0000 == 0b0000_0000 {
258        (bytes[offset], 1)
259    } else if bytes[offset] & 0b1110_0000 == 0b1100_0000 {
260        let a = (bytes[offset] & 0b0001_1111) as u16;
261        let b = (bytes[offset + 1] & 0b0011_1111) as u16;
262        let ch = a << 6 | b;
263        if ch > 0xff {
264            panic!("input string cannot be encoded as Latin-1");
265        }
266        (ch as u8, 2)
267    } else {
268        // Latin-1 code points only go up to 0xff, so if the input contains any
269        // UTF-8 characters larger than two bytes it cannot be converted to
270        // Latin-1.
271        panic!("input string cannot be encoded as Latin-1");
272    }
273}

275/// Count the number of Latin-1 characters in a string.
276///
277/// Panics if the string cannot be encoded in Latin-1.
278///
279/// This is public but hidden; it is used in the `cstr8` macro.
280#[must_use]
281pub const fn str_num_latin1_chars(s: &str) -> usize {
282    let bytes = s.as_bytes();
283    let len = bytes.len();
284
285    let mut offset = 0;
286    let mut num_latin1_chars = 0;
287
288    while offset < len {
289        // SAFETY: `bytes` is valid UTF-8.
290        let (_, num_utf8_bytes) = unsafe { latin1_from_utf8_at_offset(bytes, offset) };
291        offset += num_utf8_bytes;
292        num_latin1_chars += 1;
293    }
294
295    num_latin1_chars
296}

298/// Convert a `str` into a null-terminated Latin-1 character array.
299///
300/// Panics if the string cannot be encoded in Latin-1.
301///
302/// This is public but hidden; it is used in the `cstr8` macro.
303#[must_use]
304pub const fn str_to_latin1<const N: usize>(s: &str) -> [u8; N] {
305    let bytes = s.as_bytes();
306    let len = bytes.len();
307
308    let mut output = [0; N];
309
310    let mut output_offset = 0;
311    let mut input_offset = 0;
312    while input_offset < len {
313        // SAFETY: `bytes` is valid UTF-8.
314        let (ch, num_utf8_bytes) = unsafe { latin1_from_utf8_at_offset(bytes, input_offset) };
315        if ch == 0 {
316            panic!("interior null character");
317        } else {
318            output[output_offset] = ch;
319            output_offset += 1;
320            input_offset += num_utf8_bytes;
321        }
322    }
323
324    // The output array must be one bigger than the converted string,
325    // to leave room for the trailing null character.
326    if output_offset + 1 != N {
327        panic!("incorrect array length");
328    }
329
330    output
331}

333/// An UCS-2 null-terminated string slice.
334///
335/// This type is largely inspired by [`core::ffi::CStr`] with the exception that all characters are
336/// guaranteed to be 16 bit long.
337///
338/// For convenience, a [`CStr16`] is comparable with [`core::str`] and
339/// `alloc::string::String` from the standard library through the trait [`EqStrUntilNul`].
340#[derive(Eq, PartialEq, Ord, PartialOrd, Hash)]
341#[repr(transparent)]
342pub struct CStr16([Char16]);
343
344impl CStr16 {
345    /// Wraps a raw UEFI string with a safe C string wrapper
346    ///
347    /// # Safety
348    ///
349    /// The function will start accessing memory from `ptr` until the first
350    /// null character. It's the callers responsibility to ensure `ptr` points to
351    /// a valid string, in accessible memory.
352    #[must_use]
353    pub unsafe fn from_ptr<'ptr>(ptr: *const Char16) -> &'ptr Self {
354        let mut len = 0;
355        while *ptr.add(len) != NUL_16 {
356            len += 1
357        }
358        let ptr = ptr.cast::<u16>();
359        Self::from_u16_with_nul_unchecked(slice::from_raw_parts(ptr, len + 1))
360    }
361
362    /// Creates a `&CStr16` from a u16 slice, stopping at the first nul character.
363    ///
364    /// # Errors
365    ///
366    /// An error is returned if the slice contains invalid UCS-2 characters, or
367    /// if the slice does not contain any nul character.
368    pub fn from_u16_until_nul(codes: &[u16]) -> Result<&Self, FromSliceUntilNulError> {
369        for (pos, &code) in codes.iter().enumerate() {
370            let chr =
371                Char16::try_from(code).map_err(|_| FromSliceUntilNulError::InvalidChar(pos))?;
372            if chr == NUL_16 {
373                return Ok(unsafe { Self::from_u16_with_nul_unchecked(&codes[..=pos]) });
374            }
375        }
376        Err(FromSliceUntilNulError::NoNul)
377    }
378
379    /// Creates a `&CStr16` from a u16 slice, if the slice contains exactly
380    /// one terminating null-byte and all chars are valid UCS-2 chars.
381    pub fn from_u16_with_nul(codes: &[u16]) -> Result<&Self, FromSliceWithNulError> {
382        for (pos, &code) in codes.iter().enumerate() {
383            match code.try_into() {
384                Ok(NUL_16) => {
385                    if pos != codes.len() - 1 {
386                        return Err(FromSliceWithNulError::InteriorNul(pos));
387                    } else {
388                        return Ok(unsafe { Self::from_u16_with_nul_unchecked(codes) });
389                    }
390                }
391                Err(_) => {
392                    return Err(FromSliceWithNulError::InvalidChar(pos));
393                }
394                _ => {}
395            }
396        }
397        Err(FromSliceWithNulError::NotNulTerminated)
398    }
399
400    /// Unsafely creates a `&CStr16` from a u16 slice.
401    ///
402    /// # Safety
403    ///
404    /// It's the callers responsibility to ensure chars is a valid UCS-2
405    /// null-terminated string, with no interior null characters.
406    #[must_use]
407    pub const unsafe fn from_u16_with_nul_unchecked(codes: &[u16]) -> &Self {
408        &*(ptr::from_ref(codes) as *const Self)
409    }

411    /// Creates a `&CStr16` from a [`Char16`] slice, stopping at the first nul character.
412    ///
413    /// # Errors
414    ///
415    /// An error is returned if the slice does not contain any nul character.
416    pub fn from_char16_until_nul(chars: &[Char16]) -> Result<&Self, FromSliceUntilNulError> {
417        // Find the index of the first null char.
418        let end = chars
419            .iter()
420            .position(|c| *c == NUL_16)
421            .ok_or(FromSliceUntilNulError::NoNul)?;
422
423        // Safety: the input is nul-terminated.
424        unsafe { Ok(Self::from_char16_with_nul_unchecked(&chars[..=end])) }
425    }
426
427    /// Creates a `&CStr16` from a [`Char16`] slice, if the slice is
428    /// null-terminated and has no interior null characters.
429    pub fn from_char16_with_nul(chars: &[Char16]) -> Result<&Self, FromSliceWithNulError> {
430        // Fail early if the input is empty.
431        if chars.is_empty() {
432            return Err(FromSliceWithNulError::NotNulTerminated);
433        }
434
435        // Find the index of the first null char.
436        if let Some(null_index) = chars.iter().position(|c| *c == NUL_16) {
437            // Verify the null character is at the end.
438            if null_index == chars.len() - 1 {
439                // Safety: the input is null-terminated and has no interior nulls.
440                Ok(unsafe { Self::from_char16_with_nul_unchecked(chars) })
441            } else {
442                Err(FromSliceWithNulError::InteriorNul(null_index))
443            }
444        } else {
445            Err(FromSliceWithNulError::NotNulTerminated)
446        }
447    }

449    /// Unsafely creates a `&CStr16` from a `Char16` slice.
450    ///
451    /// # Safety
452    ///
453    /// It's the callers responsibility to ensure chars is null-terminated and
454    /// has no interior null characters.
455    #[must_use]
456    pub const unsafe fn from_char16_with_nul_unchecked(chars: &[Char16]) -> &Self {
457        let ptr: *const [Char16] = chars;
458        &*(ptr as *const Self)
459    }

461    /// Convert a [`&str`] to a `&CStr16`, backed by a buffer.
462    ///
463    /// The input string must contain only characters representable with
464    /// UCS-2, and must not contain any null characters (even at the end of
465    /// the input).
466    ///
467    /// The backing buffer must be big enough to hold the converted string as
468    /// well as a trailing null character.
469    ///
470    /// # Examples
471    ///
472    /// Convert the UTF-8 string "ABC" to a `&CStr16`:
473    ///
474    /// ```
475    /// use uefi::CStr16;
476    ///
477    /// let mut buf = [0; 4];
478    /// CStr16::from_str_with_buf("ABC", &mut buf).unwrap();
479    /// ```
480    pub fn from_str_with_buf<'a>(
481        input: &str,
482        buf: &'a mut [u16],
483    ) -> Result<&'a Self, FromStrWithBufError> {
484        let mut index = 0;
485
486        // Convert to UTF-16.
487        for c in input.encode_utf16() {
488            *buf.get_mut(index)
489                .ok_or(FromStrWithBufError::BufferTooSmall)? = c;
490            index += 1;
491        }
492
493        // Add trailing null character.
494        *buf.get_mut(index)
495            .ok_or(FromStrWithBufError::BufferTooSmall)? = 0;
496
497        // Convert from u16 to Char16. This checks for invalid UCS-2 chars and
498        // interior nulls. The NotNulTerminated case is unreachable because we
499        // just added a trailing null character.
500        Self::from_u16_with_nul(&buf[..index + 1]).map_err(|err| match err {
501            FromSliceWithNulError::InvalidChar(p) => FromStrWithBufError::InvalidChar(p),
502            FromSliceWithNulError::InteriorNul(p) => FromStrWithBufError::InteriorNul(p),
503            FromSliceWithNulError::NotNulTerminated => {
504                unreachable!()
505            }
506        })
507    }

509    /// Create a `&CStr16` from an [`UnalignedSlice`] using an aligned
510    /// buffer for storage. The lifetime of the output is tied to `buf`,
511    /// not `src`.
512    pub fn from_unaligned_slice<'buf>(
513        src: &UnalignedSlice<'_, u16>,
514        buf: &'buf mut [MaybeUninit<u16>],
515    ) -> Result<&'buf Self, UnalignedCStr16Error> {
516        // The input `buf` might be longer than needed, so get a
517        // subslice of the required length.
518        let buf = buf
519            .get_mut(..src.len())
520            .ok_or(UnalignedCStr16Error::BufferTooSmall)?;
521
522        src.copy_to_maybe_uninit(buf);
523        let buf = unsafe {
524            // Safety: `copy_buf` fully initializes the slice.
525            maybe_uninit_slice_assume_init_ref(buf)
526        };
527        Self::from_u16_with_nul(buf).map_err(|e| match e {
528            FromSliceWithNulError::InvalidChar(v) => UnalignedCStr16Error::InvalidChar(v),
529            FromSliceWithNulError::InteriorNul(v) => UnalignedCStr16Error::InteriorNul(v),
530            FromSliceWithNulError::NotNulTerminated => UnalignedCStr16Error::NotNulTerminated,
531        })
532    }

598    /// Writes each [`Char16`] as a [`char`] (4 bytes long in Rust language) into the buffer.
599    /// It is up to the implementer of [`core::fmt::Write`] to convert the char to a string
600    /// with proper encoding/charset. For example, in the case of [`alloc::string::String`]
601    /// all Rust chars (UTF-32) get converted to UTF-8.
602    ///
603    /// ## Example
604    ///
605    /// ```ignore
606    /// let firmware_vendor_c16_str: CStr16 = ...;
607    /// // crate "arrayvec" uses stack-allocated arrays for Strings => no heap allocations
608    /// let mut buf = arrayvec::ArrayString::<128>::new();
609    /// firmware_vendor_c16_str.as_str_in_buf(&mut buf);
610    /// log::info!("as rust str: {}", buf.as_str());
611    /// ```
612    ///
613    /// [`alloc::string::String`]: https://doc.rust-lang.org/nightly/alloc/string/struct.String.html
614    pub fn as_str_in_buf(&self, buf: &mut dyn core::fmt::Write) -> core::fmt::Result {
615        for c16 in self.iter() {
616            buf.write_char(char::from(*c16))?;
617        }
618        Ok(())
619    }

642impl From<&CStr16> for alloc::string::String {
643    fn from(value: &CStr16) -> Self {
644        value
645            .as_slice()
646            .iter()
647            .copied()
648            .map(u16::from)
649            .map(u32::from)
650            .map(|int| char::from_u32(int).expect("Should be encodable as UTF-8"))
651            .collect::<Self>()
652    }

655impl<StrType: AsRef<str> + ?Sized> EqStrUntilNul<StrType> for CStr16 {
656    fn eq_str_until_nul(&self, other: &StrType) -> bool {
657        let other = other.as_ref();
658
659        let any_not_equal = self
660            .iter()
661            .copied()
662            .map(char::from)
663            .zip(other.chars())
664            // This only works as CStr16 is guaranteed to have a fixed character length
665            // (unlike UTF-8 or UTF-16).
666            .take_while(|(l, r)| *l != '\0' && *r != '\0')
667            .any(|(l, r)| l != r);
668
669        !any_not_equal
670    }
671}

679/// An iterator over the [`Char16`]s in a [`CStr16`].
680#[derive(Debug)]
681pub struct CStr16Iter<'a> {
682    inner: &'a CStr16,
683    pos: usize,
684}
685
686impl<'a> Iterator for CStr16Iter<'a> {
687    type Item = &'a Char16;
688
689    fn next(&mut self) -> Option<Self::Item> {
690        if self.pos >= self.inner.0.len() - 1 {
691            None
692        } else {
693            self.pos += 1;
694            self.inner.0.get(self.pos - 1)
695        }
696    }
697}

733/// The EqStrUntilNul trait helps to compare Rust strings against UEFI string types (UCS-2 strings).
734/// The given generic implementation of this trait enables us that we only have to
735/// implement one direction (`left.eq_str_until_nul(&right)`) for each UEFI string type and we
736/// get the other direction (`right.eq_str_until_nul(&left)`) for free. Hence, the relation is
737/// reflexive.
738pub trait EqStrUntilNul<StrType: ?Sized> {
739    /// Checks if the provided Rust string `StrType` is equal to [Self] until the first null character
740    /// is found. An exception is the terminating null character of [Self] which is ignored.
741    ///
742    /// As soon as the first null character in either `&self` or `other` is found, this method returns.
743    /// Note that Rust strings are allowed to contain null bytes that do not terminate the string.
744    /// Although this is rather unusual, you can compare `"foo\0bar"` with an instance of [Self].
745    /// In that case, only `foo"` is compared against [Self] (if [Self] is long enough).
746    fn eq_str_until_nul(&self, other: &StrType) -> bool;
747}
748
749// magic implementation which transforms an existing `left.eq_str_until_nul(&right)` implementation
750// into an additional working `right.eq_str_until_nul(&left)` implementation.
751impl<StrType, C16StrType> EqStrUntilNul<C16StrType> for StrType
752where
753    StrType: AsRef<str>,
754    C16StrType: EqStrUntilNul<StrType> + ?Sized,
755{
756    fn eq_str_until_nul(&self, other: &C16StrType) -> bool {
757        // reuse the existing implementation
758        other.eq_str_until_nul(self)
759    }
760}
