3use core::ptr::{self, NonNull};

5/// Copy the bytes of `val` to `ptr`, then advance pointer to just after the
6/// newly-copied bytes.
7pub unsafe fn ptr_write_unaligned_and_add<T>(ptr: &mut *mut u8, val: T) {
8    ptr.cast::<T>().write_unaligned(val);
9    *ptr = ptr.add(size_of::<T>());
10}

12/// Convert from a `u32` to a `usize`. Panic if the input does fit. On typical
13/// targets `usize` is at least as big as `u32`, so this should never panic
14/// except on unusual targets.
15///
16/// Comparison to alternatives:
17/// * `val as usize` doesn't check that `val` actually fits in a `usize`.
18/// * `usize::try_from(val).unwrap()` doesn't work in a const context.
19pub const fn usize_from_u32(val: u32) -> usize {
20    // This is essentially the same as `usize::try_from(val).unwrap()`, but
21    // works in a `const` context on stable.
22    if size_of::<usize>() < size_of::<u32>() && val < (usize::MAX as u32) {
23        panic!("value does not fit in a usize");
24    } else {
25        val as usize
26    }
27}
28
29/// Get the raw pointer from `opt`, defaulting to `null_mut`.
30pub fn opt_nonnull_to_ptr<T>(opt: Option<NonNull<T>>) -> *mut T {
31    opt.map(NonNull::as_ptr).unwrap_or(ptr::null_mut())
32}
