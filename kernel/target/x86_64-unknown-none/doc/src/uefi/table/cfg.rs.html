3//! Configuration table utilities.
4//!
5//! The configuration table is an array of GUIDs and pointers to extra system tables.
6//!
7//! For example, it can be used to find the ACPI tables.
8//!
9//! This module contains the actual entries of the configuration table,
10//! as well as GUIDs for many known vendor tables.
11
12use crate::{guid, Guid};
13use bitflags::bitflags;
14use core::ffi::c_void;
15
16/// Contains a set of GUID / pointer for a vendor-specific table.
17///
18/// The UEFI standard guarantees each entry is unique.
19#[derive(Clone, Debug, Eq, PartialEq, Ord, PartialOrd, Hash)]
20#[repr(C)]
21pub struct ConfigTableEntry {
22    /// The GUID identifying this table.
23    pub guid: Guid,
24    /// The starting address of this table.
25    ///
26    /// Whether this is a physical or virtual address depends on the table.
27    pub address: *const c_void,
28}

50/// This table contains additional information about the UEFI implementation.
51#[repr(C)]
52#[derive(Debug)]
53pub struct PropertiesTable {
54    /// Version of the UEFI properties table.
55    ///
56    /// The only valid version currently is 0x10_000.
57    pub version: u32,
58    /// Length in bytes of this table.
59    ///
60    /// The initial version's length is 16.
61    pub length: u32,
62    /// Memory protection attributes.
63    pub memory_protection: MemoryProtectionAttribute,
64}
65
66bitflags! {
67    /// Flags describing memory protection.
68    #[repr(transparent)]
69    #[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord)]
70    pub struct MemoryProtectionAttribute: usize {
71        /// If this bit is set, then the UEFI implementation will mark pages
72        /// containing data as non-executable.
73        const NON_EXECUTABLE_DATA = 1;
74    }
75}
