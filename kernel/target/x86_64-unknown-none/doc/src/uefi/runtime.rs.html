32fn runtime_services_raw_panicking() -> NonNull<uefi_raw::table::runtime::RuntimeServices> {
33    let st = table::system_table_raw_panicking();
34    // SAFETY: valid per requirements of `set_system_table`.
35    let st = unsafe { st.as_ref() };
36    NonNull::new(st.runtime_services).expect("runtime services are not active")
37}

39/// Query the current time and date information.
40pub fn get_time() -> Result<Time> {
41    let rt = runtime_services_raw_panicking();
42    let rt = unsafe { rt.as_ref() };
43
44    let mut time = Time::invalid();
45    let time_ptr: *mut Time = &mut time;
46    unsafe { (rt.get_time)(time_ptr.cast(), ptr::null_mut()) }.to_result_with_val(|| time)

77/// Checks if a variable exists.
78///
79/// Returns `Ok(true)` if the variable exists, `Ok(false)` if the variable does
80/// not exist, or `Err` if the existence of the variable could not be determined.
81///
82/// # Errors
83///
84/// * [`Status::DEVICE_ERROR`]: variable could not be read due to a hardware error.
85/// * [`Status::SECURITY_VIOLATION`]: variable could not be read due to an
86///   authentication error.
87/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
88///   after exiting boot services.
89pub fn variable_exists(name: &CStr16, vendor: &VariableVendor) -> Result<bool> {
90    let rt = runtime_services_raw_panicking();
91    let rt = unsafe { rt.as_ref() };
92
93    let attributes = ptr::null_mut();
94    let data = ptr::null_mut();
95    let mut data_size = 0;
96
97    let status = unsafe {
98        (rt.get_variable)(
99            name.as_ptr().cast(),
100            &vendor.0,
101            attributes,
102            &mut data_size,
103            data,
104        )
105    };
106
107    match status {
108        // If the variable exists, the status will be BUFFER_TOO_SMALL because
109        // data_size is 0. Empty variables do not exist, because setting a
110        // variable with empty data deletes the variable. In other words, the
111        // status will never be SUCCESS.
112        Status::BUFFER_TOO_SMALL => Ok(true),
113        Status::NOT_FOUND => Ok(false),
114        _ => Err(Error::from(status)),
115    }
116}

118/// Gets the contents and attributes of a variable. The size of `buf` must be at
119/// least as big as the variable's size, although it can be larger.
120///
121/// On success, returns a tuple containing the variable's value (a slice of
122/// `buf`) and the variable's attributes.
123///
124/// # Errors
125///
126/// * [`Status::NOT_FOUND`]: variable was not found.
127/// * [`Status::BUFFER_TOO_SMALL`]: `buf` is not large enough. The required size
128///   will be returned in the error data.
129/// * [`Status::DEVICE_ERROR`]: variable could not be read due to a hardware error.
130/// * [`Status::SECURITY_VIOLATION`]: variable could not be read due to an
131///   authentication error.
132/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
133///   after exiting boot services.
134pub fn get_variable<'buf>(
135    name: &CStr16,
136    vendor: &VariableVendor,
137    buf: &'buf mut [u8],
138) -> Result<(&'buf mut [u8], VariableAttributes), Option<usize>> {
139    let rt = runtime_services_raw_panicking();
140    let rt = unsafe { rt.as_ref() };
141
142    let mut attributes = VariableAttributes::empty();
143    let mut data_size = buf.len();
144    let status = unsafe {
145        (rt.get_variable)(
146            name.as_ptr().cast(),
147            &vendor.0,
148            &mut attributes,
149            &mut data_size,
150            buf.as_mut_ptr(),
151        )
152    };
153
154    match status {
155        Status::SUCCESS => Ok((&mut buf[..data_size], attributes)),
156        Status::BUFFER_TOO_SMALL => Err(Error::new(status, Some(data_size))),
157        _ => Err(Error::new(status, None)),
158    }
159}

161/// Gets the contents and attributes of a variable.
162///
163/// # Errors
164///
165/// * [`Status::NOT_FOUND`]: variable was not found.
166/// * [`Status::DEVICE_ERROR`]: variable could not be read due to a hardware error.
167/// * [`Status::SECURITY_VIOLATION`]: variable could not be read due to an
168///   authentication error.
169/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
170///   after exiting boot services.
171#[cfg(feature = "alloc")]
172pub fn get_variable_boxed(
173    name: &CStr16,
174    vendor: &VariableVendor,
175) -> Result<(Box<[u8]>, VariableAttributes)> {
176    let mut out_attr = VariableAttributes::empty();
177    let get_var = |buf| {
178        get_variable(name, vendor, buf).map(|(val, attr)| {
179            // `make_boxed` expects only a DST value to be returned (`val` in
180            // this case), so smuggle the `attr` value out via a separate
181            // variable.
182            out_attr = attr;
183            val
184        })
185    };
186    #[cfg(not(feature = "unstable"))]
187    {
188        make_boxed(get_var).map(|val| (val, out_attr))
189    }
190    #[cfg(feature = "unstable")]
191    {
192        make_boxed(get_var, Global).map(|val| (val, out_attr))
193    }
194}

196/// Gets each variable key (name and vendor) one at a time.
197///
198/// This is used to iterate over variable keys. See [`variable_keys`] for a more
199/// convenient interface that requires the `alloc` feature.
200///
201/// To get the first variable key, `name` must be initialized to start with a
202/// null character. The `vendor` value is arbitrary. On success, the first
203/// variable's name and vendor will be written out to `name` and `vendor`. Keep
204/// calling `get_next_variable_key` with the same `name` and `vendor` references
205/// to get the remaining variable keys.
206///
207/// All variable names should be valid strings, but this may not be enforced by
208/// firmware. To convert to a string, truncate at the first null and call
209/// [`CStr16::from_u16_with_nul`].
210///
211/// # Errors
212///
213/// * [`Status::NOT_FOUND`]: indicates end of iteration, the last variable keys
214///   was retrieved by the previous call to `get_next_variable_key`.
215/// * [`Status::BUFFER_TOO_SMALL`]: `name` is not large enough. The required
216///   size (in `u16` characters, not bytes) will be returned in the error data.
217/// * [`Status::INVALID_PARAMETER`]: `name` does not contain a null character, or
218///   the `name` and `vendor` are not an existing variable.
219/// * [`Status::DEVICE_ERROR`]: variable could not be read due to a hardware error.
220/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
221///   after exiting boot services.
222pub fn get_next_variable_key(
223    name: &mut [u16],
224    vendor: &mut VariableVendor,
225) -> Result<(), Option<usize>> {
226    let rt = runtime_services_raw_panicking();
227    let rt = unsafe { rt.as_ref() };
228
229    let mut name_size_in_bytes = size_of_val(name);
230
231    let status = unsafe {
232        (rt.get_next_variable_name)(&mut name_size_in_bytes, name.as_mut_ptr(), &mut vendor.0)
233    };
234    match status {
235        Status::SUCCESS => Ok(()),
236        Status::BUFFER_TOO_SMALL => Err(Error::new(
237            status,
238            Some(name_size_in_bytes / size_of::<u16>()),
239        )),
240        _ => Err(Error::new(status, None)),
241    }
242}

253/// Iterator over all UEFI variables.
254///
255/// Each iteration yields a `Result<`[`VariableKey`]`>`. Error values:
256///
257/// * [`Status::DEVICE_ERROR`]: variable could not be read due to a hardware error.
258/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
259///   after exiting boot services.
260#[cfg(feature = "alloc")]
261#[derive(Debug)]
262pub struct VariableKeys {
263    name: Vec<u16>,
264    vendor: VariableVendor,
265    is_done: bool,
266}
267
268#[cfg(feature = "alloc")]
269impl VariableKeys {
270    fn new() -> Self {
271        // Create a the name buffer with a reasonable default capacity, and
272        // initialize it to an empty null-terminated string.
273        let mut name = Vec::with_capacity(32);
274        name.push(0);
275
276        Self {
277            // Give the name buffer a reasonable default capacity.
278            name,
279            // The initial vendor GUID is arbitrary.
280            vendor: VariableVendor(Guid::default()),
281            is_done: false,
282        }
283    }
284}
285
286#[cfg(feature = "alloc")]
287impl Iterator for VariableKeys {
288    type Item = Result<VariableKey>;
289
290    fn next(&mut self) -> Option<Result<VariableKey>> {
291        if self.is_done {
292            return None;
293        }
294
295        let mut result = get_next_variable_key(&mut self.name, &mut self.vendor);
296
297        // If the name buffer was too small, resize it to be big enough and call
298        // `get_next_variable_key` again.
299        if let Err(err) = &result {
300            if let Some(required_size) = err.data() {
301                self.name.resize(*required_size, 0u16);
302                result = get_next_variable_key(&mut self.name, &mut self.vendor);
303            }
304        }
305
306        match result {
307            Ok(()) => {
308                // Convert the name to a `CStr16`, yielding an error if invalid.
309                let Ok(name) = CStr16::from_u16_until_nul(&self.name) else {
310                    return Some(Err(Status::UNSUPPORTED.into()));
311                };
312
313                Some(Ok(VariableKey {
314                    name: name.to_owned(),
315                    vendor: self.vendor,
316                }))
317            }
318            Err(err) => {
319                if err.status() == Status::NOT_FOUND {
320                    // This status indicates the end of the list. The final variable
321                    // has already been yielded at this point, so return `None`.
322                    self.is_done = true;
323                    None
324                } else {
325                    // Return the error and end iteration.
326                    self.is_done = true;
327                    Some(Err(err.to_err_without_payload()))
328                }
329            }
330        }
331    }
332}

334/// Sets the value of a variable. This can be used to create a new variable,
335/// update an existing variable, or (when the size of `data` is zero)
336/// delete a variable.
337///
338/// # Warnings
339///
340/// The [`Status::WARN_RESET_REQUIRED`] warning will be returned when using
341/// this function to transition the Secure Boot mode to setup mode or audit
342/// mode if the firmware requires a reboot for that operation.
343///
344/// # Errors
345///
346/// * [`Status::INVALID_PARAMETER`]: invalid attributes, name, or vendor.
347/// * [`Status::OUT_OF_RESOURCES`]: not enough storage is available to hold
348///   the variable.
349/// * [`Status::WRITE_PROTECTED`]: variable is read-only.
350/// * [`Status::SECURITY_VIOLATION`]: variable could not be written due to an
351///   authentication error.
352/// * [`Status::NOT_FOUND`]: attempted to update a non-existent variable.
353/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
354///   after exiting boot services.
355pub fn set_variable(
356    name: &CStr16,
357    vendor: &VariableVendor,
358    attributes: VariableAttributes,
359    data: &[u8],
360) -> Result {
361    let rt = runtime_services_raw_panicking();
362    let rt = unsafe { rt.as_ref() };
363
364    unsafe {
365        (rt.set_variable)(
366            name.as_ptr().cast(),
367            &vendor.0,
368            attributes,
369            data.len(),
370            data.as_ptr(),
371        )
372        .to_result()
373    }
374}

376/// Deletes a UEFI variable.
377///
378/// # Errors
379///
380/// * [`Status::INVALID_PARAMETER`]: invalid name or vendor.
381/// * [`Status::WRITE_PROTECTED`]: variable is read-only.
382/// * [`Status::SECURITY_VIOLATION`]: variable could not be deleted due to an
383///   authentication error.
384/// * [`Status::NOT_FOUND`]: attempted to delete a non-existent variable.
385/// * [`Status::UNSUPPORTED`]: this platform does not support variable storage
386///   after exiting boot services.
387pub fn delete_variable(name: &CStr16, vendor: &VariableVendor) -> Result {
388    set_variable(name, vendor, VariableAttributes::empty(), &[])
389}

391/// Get information about UEFI variable storage space for the type
392/// of variable specified in `attributes`.
393///
394/// This operation is only supported starting with UEFI 2.0.
395///
396/// See [`VariableStorageInfo`] for details of the information returned.
397///
398/// # Errors
399///
400/// * [`Status::INVALID_PARAMETER`]: invalid combination of variable attributes.
401/// * [`Status::UNSUPPORTED`]: the combination of variable attributes is not
402///   supported on this platform, or the UEFI version is less than 2.0.
403pub fn query_variable_info(attributes: VariableAttributes) -> Result<VariableStorageInfo> {
404    let rt = runtime_services_raw_panicking();
405    let rt = unsafe { rt.as_ref() };
406
407    if rt.header.revision < Revision::EFI_2_00 {
408        return Err(Status::UNSUPPORTED.into());
409    }
410
411    let mut info = VariableStorageInfo::default();
412    unsafe {
413        (rt.query_variable_info)(
414            attributes,
415            &mut info.maximum_variable_storage_size,
416            &mut info.remaining_variable_storage_size,
417            &mut info.maximum_variable_size,
418        )
419        .to_result_with_val(|| info)
420    }
421}

423/// Passes capsules to the firmware.
424///
425/// Capsules are most commonly used to update system firmware.
426///
427/// # Errors
428///
429/// * [`Status::INVALID_PARAMETER`]: zero capsules were provided, or the
430///   capsules are invalid.
431/// * [`Status::DEVICE_ERROR`]: the capsule update was started but failed to a
432///   device error.
433/// * [`Status::OUT_OF_RESOURCES`]: before exiting boot services, indicates the
434///   capsule is compatible with the platform but there are insufficient
435///   resources to complete the update. After exiting boot services, indicates
436///   the capsule is compatible with the platform but can only be processed
437///   before exiting boot services.
438/// * [`Status::UNSUPPORTED`]: this platform does not support capsule updates
439///   after exiting boot services.
440pub fn update_capsule(
441    capsule_header_array: &[&CapsuleHeader],
442    capsule_block_descriptors: &[CapsuleBlockDescriptor],
443) -> Result {
444    let rt = runtime_services_raw_panicking();
445    let rt = unsafe { rt.as_ref() };
446
447    unsafe {
448        (rt.update_capsule)(
449            capsule_header_array.as_ptr().cast(),
450            capsule_header_array.len(),
451            capsule_block_descriptors.as_ptr() as PhysicalAddress,
452        )
453        .to_result()
454    }
455}

487/// Resets the computer.
488///
489/// See [`ResetType`] for details of the various reset types.
490///
491/// For a normal reset the value of `status` should be
492/// [`Status::SUCCESS`]. Otherwise, an error code can be used.
493///
494/// The `data` arg is usually `None`. Otherwise, it must contain a UCS-2
495/// null-terminated string followed by additional binary data. For
496/// [`ResetType::PLATFORM_SPECIFIC`], the binary data must be a vendor-specific
497/// [`Guid`] that indicates the type of reset to perform.
498///
499/// This function never returns.
500pub fn reset(reset_type: ResetType, status: Status, data: Option<&[u8]>) -> ! {
501    let rt = runtime_services_raw_panicking();
502    let rt = unsafe { rt.as_ref() };
503
504    let (size, data) = data
505        .map(|data| (data.len(), data.as_ptr()))
506        .unwrap_or((0, ptr::null()));
507
508    unsafe { (rt.reset_system)(reset_type, status, size, data) }
509}

511/// Changes the runtime addressing mode of EFI firmware from physical to
512/// virtual. It is up to the caller to translate the old system table address
513/// to a new virtual address and provide it for this function.
514///
515/// If successful, this function will call [`set_system_table`] with
516/// `new_system_table_virtual_addr`.
517///
518/// [`set_system_table`]: table::set_system_table
519///
520/// # Safety
521///
522/// The caller must ensure the memory map is valid.
523///
524/// # Errors
525///
526/// * [`Status::UNSUPPORTED`]: either boot services haven't been exited, the
527///   firmware's addressing mode is already virtual, or the firmware does not
528///   support this operation.
529/// * [`Status::NO_MAPPING`]: `map` is missing a required range.
530/// * [`Status::NOT_FOUND`]: `map` contains an address that is not in the
531///   current memory map.
532pub unsafe fn set_virtual_address_map(
533    map: &mut [MemoryDescriptor],
534    new_system_table_virtual_addr: *const uefi_raw::table::system::SystemTable,
535) -> Result {
536    let rt = runtime_services_raw_panicking();
537    let rt = unsafe { rt.as_ref() };
538
539    // Unsafe Code Guidelines guarantees that there is no padding in an array or a slice
540    // between its elements if the element type is `repr(C)`, which is our case.
541    //
542    // See https://rust-lang.github.io/unsafe-code-guidelines/layout/arrays-and-slices.html
543    let map_size = size_of_val(map);
544    let entry_size = size_of::<MemoryDescriptor>();
545    let entry_version = MemoryDescriptor::VERSION;
546    let map_ptr = map.as_mut_ptr();
547    (rt.set_virtual_address_map)(map_size, entry_size, entry_version, map_ptr).to_result()?;
548
549    // Update the global system table pointer.
550    table::set_system_table(new_system_table_virtual_addr);
551
552    Ok(())
553}

611impl Display for TimeError {
612    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
613        if self.year {
614            writeln!(f, "year not within `1900..=9999`")?;
615        }
616        if self.month {
617            writeln!(f, "month not within `1..=12")?;
618        }
619        if self.day {
620            writeln!(f, "day not within `1..=31`")?;
621        }
622        if self.hour {
623            writeln!(f, "hour not within `0..=23`")?;
624        }
625        if self.minute {
626            writeln!(f, "minute not within `0..=59`")?;
627        }
628        if self.second {
629            writeln!(f, "second not within `0..=59`")?;
630        }
631        if self.nanosecond {
632            writeln!(f, "nanosecond not within `0..=999_999_999`")?;
633        }
634        if self.timezone {
635            writeln!(
636                f,
637                "time_zone not `Time::UNSPECIFIED_TIMEZONE` nor within `-1440..=1440`"
638            )?;
639        }
640        if self.daylight {
641            writeln!(f, "unknown bits set for daylight")?;
642        }
643        Ok(())
644    }
645}

647impl Time {
648    /// Unspecified Timezone/local time.
649    const UNSPECIFIED_TIMEZONE: i16 = uefi_raw::time::Time::UNSPECIFIED_TIMEZONE;
650
651    /// Create a `Time` value. If a field is not in the valid range,
652    /// [`TimeError`] is returned.
653    pub fn new(params: TimeParams) -> core::result::Result<Self, TimeError> {
654        let time = Self(uefi_raw::time::Time {
655            year: params.year,
656            month: params.month,
657            day: params.day,
658            hour: params.hour,
659            minute: params.minute,
660            second: params.second,
661            pad1: 0,
662            nanosecond: params.nanosecond,
663            time_zone: params.time_zone.unwrap_or(Self::UNSPECIFIED_TIMEZONE),
664            daylight: params.daylight,
665            pad2: 0,
666        });
667
668        time.is_valid().map(|_| time)
669    }

672    /// be used with [`FileInfo`] to indicate a field should not be
673    /// updated when calling [`File::set_info`].
674    ///
675    /// [`FileInfo`]: uefi::proto::media::file::FileInfo
676    /// [`File::set_info`]: uefi::proto::media::file::File::set_info
677    #[must_use]
678    pub const fn invalid() -> Self {
679        Self(uefi_raw::time::Time::invalid())
680    }
681
682    /// `Ok()` if all fields are within valid ranges, `Err(TimeError)` otherwise.
683    pub fn is_valid(&self) -> core::result::Result<(), TimeError> {
684        let mut err = TimeError::default();
685        if !(1900..=9999).contains(&self.year()) {
686            err.year = true;
687        }
688        if !(1..=12).contains(&self.month()) {
689            err.month = true;
690        }
691        if !(1..=31).contains(&self.day()) {
692            err.day = true;
693        }
694        if self.hour() > 23 {
695            err.hour = true;
696        }
697        if self.minute() > 59 {
698            err.minute = true;
699        }
700        if self.second() > 59 {
701            err.second = true;
702        }
703        if self.nanosecond() > 999_999_999 {
704            err.nanosecond = true;
705        }
706        if self.time_zone().is_some() && !((-1440..=1440).contains(&self.time_zone().unwrap())) {
707            err.timezone = true;
708        }
709        // All fields are false, i.e., within their valid range.
710        if err == TimeError::default() {
711            Ok(())
712        } else {
713            Err(err)
714        }
715    }

803/// Error returned from failing to convert a byte slice into a [`Time`].
804#[derive(Clone, Copy, Debug, PartialEq, Eq)]
805pub enum TimeByteConversionError {
806    /// One or more fields of the converted [`Time`] is invalid.
807    InvalidFields(TimeError),
808    /// The byte slice is not large enough to hold a [`Time`].
809    InvalidSize,
810}
811
812impl Display for TimeByteConversionError {
813    fn fmt(&self, f: &mut Formatter) -> fmt::Result {
814        match self {
815            Self::InvalidFields(error) => write!(f, "{error}"),
816            Self::InvalidSize => write!(
817                f,
818                "the byte slice is not large enough to hold a Time struct"
819            ),
820        }
821    }
822}
823
824impl TryFrom<&[u8]> for Time {
825    type Error = TimeByteConversionError;
826
827    fn try_from(bytes: &[u8]) -> core::result::Result<Self, Self::Error> {
828        if size_of::<Self>() <= bytes.len() {
829            let year = u16::from_le_bytes(bytes[0..2].try_into().unwrap());
830            let month = bytes[2];
831            let day = bytes[3];
832            let hour = bytes[4];
833            let minute = bytes[5];
834            let second = bytes[6];
835            let nanosecond = u32::from_le_bytes(bytes[8..12].try_into().unwrap());
836            let time_zone = match i16::from_le_bytes(bytes[12..14].try_into().unwrap()) {
837                Self::UNSPECIFIED_TIMEZONE => None,
838                num => Some(num),
839            };
840            let daylight = Daylight::from_bits(bytes[14]).ok_or(
841                TimeByteConversionError::InvalidFields(TimeError {
842                    daylight: true,
843                    ..Default::default()
844                }),
845            )?;
846
847            let time_params = TimeParams {
848                year,
849                month,
850                day,
851                hour,
852                minute,
853                second,
854                nanosecond,
855                time_zone,
856                daylight,
857            };
858
859            Self::new(time_params).map_err(TimeByteConversionError::InvalidFields)
860        } else {
861            Err(TimeByteConversionError::InvalidSize)
862        }
863    }
864}
