3//! Module for the traits [`MemoryMap`] and [`MemoryMapMut`].
4
5use super::*;
6use core::fmt::Debug;
7use core::ops::{Index, IndexMut};

9/// An accessory to the UEFI memory map and associated metadata that can be
10/// either iterated or indexed like an array.
11///
12/// A [`MemoryMap`] is always associated with the unique [`MemoryMapKey`]
13/// bundled with the map.
14///
15/// To iterate over the entries, call [`MemoryMap::entries`].
16///
17/// ## UEFI pitfalls
18/// Note that a MemoryMap can quickly become outdated, as soon as any explicit
19/// or hidden allocation happens.
20///
21/// As soon as boot services are excited, all previous obtained memory maps must
22/// be considered as outdated, except if the [`MemoryMapKey`] equals the one
23/// returned by `exit_boot_services()`.
24///
25/// **Please note** that when working with memory maps, the `entry_size` is
26/// usually larger than `size_of::<MemoryDescriptor` [[0]]. So to be safe,
27/// always use `entry_size` as step-size when interfacing with the memory map on
28/// a low level.
29///
30/// [0]: https://github.com/tianocore/edk2/blob/7142e648416ff5d3eac6c6d607874805f5de0ca8/MdeModulePkg/Core/PiSmmCore/Page.c#L1059
31pub trait MemoryMap: Debug + Index<usize, Output = MemoryDescriptor> {
32    /// Returns the associated [`MemoryMapMeta`].
33    #[must_use]
34    fn meta(&self) -> MemoryMapMeta;
35
36    /// Returns the associated [`MemoryMapKey`]. Note that this isn't
37    /// necessarily the key of the latest valid UEFI memory map.
38    #[must_use]
39    fn key(&self) -> MemoryMapKey;
40
41    /// Returns the number of keys in the map.
42    #[must_use]
43    fn len(&self) -> usize;
44
45    /// Returns if the memory map is empty.
46    #[must_use]
47    fn is_empty(&self) -> bool {
48        self.len() == 0
49    }
50
51    /// Returns a reference to the [`MemoryDescriptor`] at the given index, if
52    /// present.
53    #[must_use]
54    fn get(&self, index: usize) -> Option<&MemoryDescriptor> {
55        if index >= self.len() {
56            None
57        } else {
58            let offset = index * self.meta().desc_size;
59            unsafe {
60                self.buffer()
61                    .as_ptr()
62                    .add(offset)
63                    .cast::<MemoryDescriptor>()
64                    .as_ref()
65            }
66        }
67    }
68
69    /// Returns a reference to the underlying memory.
70    #[must_use]
71    fn buffer(&self) -> &[u8];
72
73    /// Returns an Iterator of type [`MemoryMapIter`].
74    #[must_use]
75    fn entries(&self) -> MemoryMapIter<'_>;
76
77    /// Returns if the underlying memory map is sorted regarding the physical
78    /// address start.
79    #[must_use]
80    fn is_sorted(&self) -> bool {
81        let iter = self.entries();
82        let iter = iter.clone().zip(iter.skip(1));
83
84        for (curr, next) in iter {
85            if next.phys_start < curr.phys_start {
86                log::debug!("next.phys_start < curr.phys_start: curr={curr:?}, next={next:?}");
87                return false;
88            }
89        }
90        true
91    }
92}
93
94/// Extension to [`MemoryMap`] that adds mutable operations. This also includes
95/// the ability to sort the memory map.
96pub trait MemoryMapMut: MemoryMap + IndexMut<usize> {
97    /// Returns a mutable reference to the [`MemoryDescriptor`] at the given
98    /// index, if present.
99    #[must_use]
100    fn get_mut(&mut self, index: usize) -> Option<&mut MemoryDescriptor> {
101        if index >= self.len() {
102            None
103        } else {
104            let offset = index * self.meta().desc_size;
105            unsafe {
106                self.buffer_mut()
107                    .as_mut_ptr()
108                    .add(offset)
109                    .cast::<MemoryDescriptor>()
110                    .as_mut()
111            }
112        }
113    }
114
115    /// Sorts the memory map by physical address in place. This operation is
116    /// optional and should be invoked only once.
117    fn sort(&mut self);
118
119    /// Returns a reference to the underlying memory.
120    ///
121    /// # Safety
122    ///
123    /// This is unsafe as there is a potential to create invalid entries.
124    unsafe fn buffer_mut(&mut self) -> &mut [u8];
125}
