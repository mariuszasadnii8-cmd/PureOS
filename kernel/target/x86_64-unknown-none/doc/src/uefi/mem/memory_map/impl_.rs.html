14/// Errors that may happen when constructing a [`MemoryMapRef`] or
15/// [`MemoryMapRefMut`].
16#[derive(Copy, Clone, Debug)]

20    /// The memory map size is invalid.

32/// Implementation of [`MemoryMap`] for the given buffer.
33#[derive(Debug)]
34pub struct MemoryMapRef<'a> {
35    buf: &'a [u8],
36    meta: MemoryMapMeta,
37    len: usize,
38}
39
40impl<'a> MemoryMapRef<'a> {
41    /// Constructs a new [`MemoryMapRef`].
42    ///
43    /// The underlying memory might contain an invalid/malformed memory map
44    /// which can't be checked during construction of this type. The entry
45    /// iterator might yield unexpected results.
46    pub fn new(buffer: &'a [u8], meta: MemoryMapMeta) -> Result<Self, MemoryMapError> {
47        if buffer.as_ptr().align_offset(8) != 0 {
48            return Err(MemoryMapError::Misaligned);
49        }
50        if buffer.len() < meta.map_size {
51            return Err(MemoryMapError::InvalidSize);
52        }
53        Ok(Self {
54            buf: buffer,
55            meta,
56            len: meta.entry_count(),
57        })
58    }
59}

158impl MemoryMapRefMut<'_> {
159    /// Hoare partition scheme for quicksort.
160    /// Must be called with `low` and `high` being indices within bounds.
161    fn qsort(&mut self, low: usize, high: usize) {
162        if low >= high {
163            return;
164        }
165
166        let p = self.partition(low, high);
167        self.qsort(low, p);
168        self.qsort(p + 1, high);
169    }
170
171    fn partition(&mut self, low: usize, high: usize) -> usize {
172        let pivot = self.get_element_phys_addr(low + (high - low) / 2);
173
174        let mut left_index = low.wrapping_sub(1);
175        let mut right_index = high.wrapping_add(1);
176
177        loop {
178            while {
179                left_index = left_index.wrapping_add(1);
180
181                self.get_element_phys_addr(left_index) < pivot
182            } {}
183
184            while {
185                right_index = right_index.wrapping_sub(1);
186
187                self.get_element_phys_addr(right_index) > pivot
188            } {}
189
190            if left_index >= right_index {
191                return right_index;
192            }
193
194            self.swap(left_index, right_index);
195        }
196    }
197
198    /// Indices must be smaller than len.
199    fn swap(&mut self, index1: usize, index2: usize) {
200        assert!(index1 < self.len);
201        assert!(index2 < self.len);
202
203        if index1 == index2 {
204            return;
205        }
206
207        let base = self.buf.as_mut_ptr();
208
209        let offset1 = index1 * self.meta.desc_size;
210        let offset2 = index2 * self.meta.desc_size;
211
212        // SAFETY: the data starting at `offset1` and `offset2` are valid
213        // descriptors, and do not overlap.
214        unsafe {
215            ptr::swap_nonoverlapping(base.add(offset1), base.add(offset2), self.meta.desc_size);
216        }
217    }
218
219    fn get_element_phys_addr(&self, index: usize) -> PhysicalAddress {
220        assert!(index < self.len);
221        let offset = index.checked_mul(self.meta.desc_size).unwrap();
222        // SAFETY: the data starting at `offset` is a valid descriptor.
223        let elem = unsafe { &*self.buf.as_ptr().add(offset).cast::<MemoryDescriptor>() };
224        elem.phys_start
225    }
226}

242/// The backing memory for the UEFI memory app on the UEFI heap, allocated using
243/// the UEFI boot services allocator. This occupied memory will also be
244/// reflected in the memory map itself.
245///
246/// Although untyped, it is similar to the `Box` type in terms of heap
247/// allocation and deallocation, as well as ownership of the corresponding
248/// memory. Apart from that, this type only has the semantics of a buffer.
249///
250/// The memory is untyped, which is necessary due to the nature of the UEFI
251/// spec. It still ensures a correct alignment to hold [`MemoryDescriptor`]. The
252/// size of the buffer is sufficient to hold the memory map at the point in time
253/// where this is created. Note that due to (not obvious or asynchronous)
254/// allocations/deallocations in your environment, this might be outdated at the
255/// time you store the memory map in it.
256///
257/// Note that due to the nature of the UEFI memory app, this buffer might
258/// hold (a few) bytes more than necessary. The `map_size` reported by
259/// `get_memory_map` tells the actual size.
260///
261/// When this type is dropped and boot services are not exited yet, the memory
262/// is freed.
263///
264/// # Usage
265/// The type is intended to be used like this:
266/// 1. create it using [`MemoryMapBackingMemory::new`]
267/// 2. pass it to [`boot::get_memory_map`]
268/// 3. construct a [`MemoryMapOwned`] from it
269///
270/// [`boot::get_memory_map`]: crate::boot::get_memory_map
271#[derive(Debug)]
272#[allow(clippy::len_without_is_empty)] // this type is never empty
273pub(crate) struct MemoryMapBackingMemory(NonNull<[u8]>);
274
275impl MemoryMapBackingMemory {
276    /// Constructs a new [`MemoryMapBackingMemory`].
277    ///
278    /// # Parameters
279    /// - `memory_type`: The memory type for the memory map allocation.
280    ///   Typically, [`MemoryType::LOADER_DATA`] for regular UEFI applications.
281    pub(crate) fn new(memory_type: MemoryType) -> crate::Result<Self> {
282        let memory_map_meta = boot::memory_map_size();
283        let len = Self::safe_allocation_size_hint(memory_map_meta);
284        let ptr = boot::allocate_pool(memory_type, len)?.as_ptr();
285
286        // Should be fine as UEFI always has  allocations with a guaranteed
287        // alignment of 8 bytes.
288        assert_eq!(ptr.align_offset(align_of::<MemoryDescriptor>()), 0);
289
290        // If this panics, the UEFI implementation is broken.
291        assert_eq!(memory_map_meta.map_size % memory_map_meta.desc_size, 0);
292
293        unsafe { Ok(Self::from_raw(ptr, len)) }
294    }
295
296    unsafe fn from_raw(ptr: *mut u8, len: usize) -> Self {
297        assert_eq!(ptr.align_offset(align_of::<MemoryDescriptor>()), 0);
298
299        let ptr = NonNull::new(ptr).expect("UEFI should never return a null ptr. An error should have been reflected via an Err earlier.");
300        let slice = NonNull::slice_from_raw_parts(ptr, len);
301
302        Self(slice)
303    }
304
305    /// INTERNAL, for unit tests.
306    ///
307    /// Creates an instance from the provided memory, which is not necessarily
308    /// on the UEFI heap.
309    #[cfg(test)]
310    pub(crate) fn from_slice(buffer: &mut [u8]) -> Self {
311        let len = buffer.len();
312        unsafe { Self::from_raw(buffer.as_mut_ptr(), len) }
313    }
314
315    /// Returns a "safe" best-effort size hint for the memory map size with
316    /// some additional bytes in buffer compared to the [`MemoryMapMeta`]. This
317    /// takes into account that, as you go, more (small) allocations might
318    /// happen.
319    #[must_use]
320    const fn safe_allocation_size_hint(mmm: MemoryMapMeta) -> usize {
321        // Allocate space for extra entries beyond the current size of the
322        // memory map. The value of 8 matches the value in the Linux kernel:
323        // https://github.com/torvalds/linux/blob/e544a07438/drivers/firmware/efi/libstub/efistub.h#L173
324        const EXTRA_ENTRIES: usize = 8;
325
326        let extra_size = mmm.desc_size * EXTRA_ENTRIES;
327        mmm.map_size + extra_size
328    }
329
330    /// Returns a slice to the underlying memory.
331    #[must_use]
332    pub const fn as_slice(&self) -> &[u8] {
333        unsafe { self.0.as_ref() }
334    }
335
336    /// Returns a mutable slice to the underlying memory.
337    #[must_use]
338    pub fn as_mut_slice(&mut self) -> &mut [u8] {
339        unsafe { self.0.as_mut() }
340    }
341}
342
343// Don't drop when we use this in unit tests.
344impl Drop for MemoryMapBackingMemory {
345    fn drop(&mut self) {
346        if boot::are_boot_services_active() {
347            let res = unsafe { boot::free_pool(self.0.cast()) };
348            if let Err(e) = res {
349                log::error!("Failed to deallocate memory map: {e:?}");
350            }
351        } else {
352            log::debug!("Boot services are exited. Memory map won't be freed using the UEFI boot services allocator.");
353        }
354    }
355}

357/// Implementation of [`MemoryMapMut`] that owns the buffer on the UEFI heap.
358#[derive(Debug)]
359pub struct MemoryMapOwned {
360    /// Backing memory, properly initialized at this point.
361    pub(crate) buf: MemoryMapBackingMemory,
362    pub(crate) meta: MemoryMapMeta,
363    pub(crate) len: usize,
364}
365
366impl MemoryMapOwned {
367    /// Creates a [`MemoryMapOwned`] from the given **initialized** memory map
368    /// (stored inside the provided buffer) and the corresponding
369    /// [`MemoryMapMeta`].
370    pub(crate) fn from_initialized_mem(buf: MemoryMapBackingMemory, meta: MemoryMapMeta) -> Self {
371        assert!(meta.desc_size >= size_of::<MemoryDescriptor>());
372        let len = meta.entry_count();
373        Self { buf, meta, len }
374    }
375}

466    fn mmap_raw<'a>(memory: &mut [MemoryDescriptor]) -> (&'a mut [u8], MemoryMapMeta) {
467        let desc_size = size_of::<MemoryDescriptor>();
468        let len = memory.len() * desc_size;
469        let ptr = memory.as_mut_ptr().cast::<u8>();
470        let slice = unsafe { core::slice::from_raw_parts_mut(ptr, len) };
471        let meta = MemoryMapMeta {
472            map_size: len,
473            desc_size,
474            map_key: Default::default(),
475            desc_version: MemoryDescriptor::VERSION,
476        };
477        (slice, meta)
478    }
